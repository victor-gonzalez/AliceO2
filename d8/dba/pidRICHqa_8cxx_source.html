77  template <typename T>
78  void makelogaxis(T h)
79  {
80  if (logAxis == 0) {
81  return;
82  }
83  const int nbins = h->GetNbinsX();
84  double binp[nbins + 1];
85  double max = h->GetXaxis()->GetBinUpEdge(nbins);
86  double min = h->GetXaxis()->GetBinLowEdge(1);
87  if (min <= 0) {
88  min = 0.00001;
89  }
90  double lmin = TMath::Log10(min);
91  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
92  for (int i = 0; i < nbins; i++) {
93  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
94  }
95  binp[nbins] = max + 1;
96  h->GetXaxis()->Set(nbins, binp);
97  }

164  template <uint8_t pidIndex, typename T, typename TTT, typename TT>
165  void fillNsigma(const T& track, const TTT& particle, const TT& mcParticles, const float& nsigma)
166  {
167  if (abs(particle.pdgCode()) == PDGs[pidIndex]) {
168  histos.fill(HIST(hnsigmaMC[pidIndex]), track.pt(), nsigma);
169 
170  if (MC::isPhysicalPrimary(mcParticles, particle)) { // Selecting primaries
171  histos.fill(HIST(hnsigmaMCprm[pidIndex]), track.pt(), nsigma);
172  } else {
173  histos.fill(HIST(hnsigmaMCsec[pidIndex]), track.pt(), nsigma);
174  }
175  }
176  }
177 
178  using Trks = soa::Join<aod::Tracks, aod::RICHTracksIndex, aod::TracksExtra,
179  aod::pidTOFFullEl, aod::pidTOFFullMu, aod::pidTOFFullPi,
180  aod::pidTOFFullKa, aod::pidTOFFullPr>;
181  void process(const Trks& tracks,
182  const aod::McTrackLabels& labels,
183  const aod::RICHs&,
184  const aod::McParticles& mcParticles,
185  const aod::Collisions& colls)
186  {
187  for (const auto& col : colls) {
188  histos.fill(HIST("event/vertexz"), col.posZ());
189  }
190  for (const auto& track : tracks) {
191  if (!track.has_rich()) {
192  continue;
193  }
194  if (track.length() < minLength) {
195  continue;
196  }
197  if (track.length() > maxLength) {
198  continue;
199  }
200  if (track.eta() > maxEta || track.eta() < minEta) {
201  continue;
202  }
203  const auto mcParticle = labels.iteratorAt(track.globalIndex()).mcParticle();
204  if (pdgCode != 0 && abs(mcParticle.pdgCode()) != pdgCode) {
205  continue;
206  }
207  if (useTOF && !track.hasTOF()) {
208  continue;
209  }
210 
211  histos.fill(HIST("p/Unselected"), track.p());
212  histos.fill(HIST("pt/Unselected"), track.pt());
213  histos.fill(HIST("qa/eta"), track.eta());
214  histos.fill(HIST("qa/signal"), track.rich().richSignal());
215  histos.fill(HIST("qa/signalerror"), track.rich().richSignalError());
216  histos.fill(HIST("qa/signalvsP"), track.p(), track.rich().richSignal());
217 
218  float delta = -999.f;
219  float nsigma = -999.f;
220  if constexpr (pid_type == 0) {
221  delta = track.rich().richDeltaEl();
222  nsigma = track.rich().richNsigmaEl();
223  if (useTOF && abs(track.tofNSigmaEl()) > 3.f) {
224  continue;
225  }
226  } else if constexpr (pid_type == 1) {
227  delta = track.rich().richDeltaMu();
228  nsigma = track.rich().richNsigmaMu();
229  if (useTOF && abs(track.tofNSigmaMu()) > 3.f) {
230  continue;
231  }
232  } else if constexpr (pid_type == 2) {
233  delta = track.rich().richDeltaPi();
234  nsigma = track.rich().richNsigmaPi();
235  if (useTOF && abs(track.tofNSigmaPi()) > 3.f) {
236  continue;
237  }
238  } else if constexpr (pid_type == 3) {
239  delta = track.rich().richDeltaKa();
240  nsigma = track.rich().richNsigmaKa();
241  if (useTOF && abs(track.tofNSigmaKa()) > 3.f) {
242  continue;
243  }
244  } else if constexpr (pid_type == 4) {
245  delta = track.rich().richDeltaPr();
246  nsigma = track.rich().richNsigmaPr();
247  if (useTOF && abs(track.tofNSigmaPr()) > 3.f) {
248  continue;
249  }
250  }
251  histos.fill(HIST(hnsigma[pid_type]), track.pt(), nsigma);
252  histos.fill(HIST(hdelta[pid_type]), track.p(), delta);
253  if (MC::isPhysicalPrimary(mcParticles, mcParticle)) { // Selecting primaries
254  histos.fill(HIST(hnsigmaprm[pid_type]), track.pt(), nsigma);
255  histos.fill(HIST("p/Prim"), track.p());
256  } else {
257  histos.fill(HIST(hnsigmasec[pid_type]), track.pt(), nsigma);
258  histos.fill(HIST("p/Sec"), track.p());
259  }
260 
261  fillNsigma<0>(track, mcParticle, mcParticles, nsigma);
262  fillNsigma<1>(track, mcParticle, mcParticles, nsigma);
263  fillNsigma<2>(track, mcParticle, mcParticles, nsigma);
264  fillNsigma<3>(track, mcParticle, mcParticles, nsigma);
265  fillNsigma<4>(track, mcParticle, mcParticles, nsigma);
266  }
267  }
268 };
