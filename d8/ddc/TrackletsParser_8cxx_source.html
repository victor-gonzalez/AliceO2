30 inline void TrackletsParser::swapByteOrder(unsigned int& ui)
31 {
32  ui = (ui >> 24) |
33  ((ui << 8) & 0x00FF0000) |
34  ((ui >> 8) & 0x0000FF00) |
35  (ui << 24);
36 }
