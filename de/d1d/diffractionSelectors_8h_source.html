37  LOGF(info, "Collision %f BC %i", collision.collisionTime(), bc.globalBC());
38  LOGF(info, "Number of close BCs: %i", bcs.size());
39  LOGF(info, "Number of tracks: %i", tracks.size());
40 
41  // characteristics of collision and nominal bc
42  // Decisions which do NOT need pre- post-pileup protection
43 
44  // Number of tracks
45  //if (tracks.size() < diffCuts.minNTracks() || tracks.size() > diffCuts.maxNTracks()) {
46  // return false;

49  // all tracks must be global tracks
50  // ATTENTION: isGlobalTrack is created by o2-analysis-trackselection
51  // The trackselection currently has fixed settings and can not be configured with
52  // command line options! This is not what we want!
53  LOGF(info, "Track types");
54  for (auto& track : tracks) {
55  LOGF(info, " : %i / %i / %i / %i", track.trackType(), track.isGlobalTrack(), track.isGlobalTrackSDD(), track.itsClusterMap());
56  }
57 
58  // tracks with TOF hits
59  // ATTENTION: in Run 2 converted data hasTOF is often true although the tofSignal
60  // has a dummy value (99998.) This will be corrected in new versions
61  // of the data - check this!
62  LOGF(info, "TOF");
63  int nTracksWithTOFHit = 0;
64  for (auto& track : tracks) {
65  LOGF(info, " [%i] signal %f / chi2: %f", track.hasTOF(), track.tofSignal(), track.tofChi2());
66  }
67  LOGF(info, "Tracks with TOF hit %i", nTracksWithTOFHit);
68 
69  // only tracks with good PID
70  for (auto& track : tracks) {
71  auto goodPID = hasGoodPID(diffCuts, track);
72  }
73 
74  // Decisions which need past-future protection
75  // This applies to 'slow' detectors
76  // loop over all selected BCs
77  LOGF(info, "Number of close BCs: %i", bcs.size());

79  LOGF(info, "BC %i / %i", bc.globalBC(), bc.triggerMask());

130 // function to check if track provides good PID information
131 // Checks the nSigma for any particle assumption to be within limits.
132 template <typename TC>
133 bool hasGoodPID(cutHolder diffCuts, TC track)
134 {
135  // El, Mu, Pi, Ka, and Pr are considered
136  // at least one nSigma must be within set limits
137  LOGF(info, "TPC PID %f / %f / %f / %f / %f",
138  track.tpcNSigmaEl(),
139  track.tpcNSigmaMu(),
140  track.tpcNSigmaPi(),
141  track.tpcNSigmaKa(),
142  track.tpcNSigmaPr());
143  if (TMath::Abs(track.tpcNSigmaEl()) < diffCuts.maxnSigmaTPC()) {
144  return true;
145  }
146  if (TMath::Abs(track.tpcNSigmaMu()) < diffCuts.maxnSigmaTPC()) {
147  return true;
148  }
149  if (TMath::Abs(track.tpcNSigmaPi()) < diffCuts.maxnSigmaTPC()) {
150  return true;
151  }
152  if (TMath::Abs(track.tpcNSigmaKa()) < diffCuts.maxnSigmaTPC()) {
153  return true;
154  }
155  if (TMath::Abs(track.tpcNSigmaPr()) < diffCuts.maxnSigmaTPC()) {
156  return true;
157  }
158 
159  if (track.hasTOF()) {
160  LOGF(info, "TOF PID %f / %f / %f / %f / %f",
161  track.tofNSigmaEl(),
162  track.tofNSigmaMu(),
163  track.tofNSigmaPi(),
164  track.tofNSigmaKa(),
165  track.tofNSigmaPr());
166  if (TMath::Abs(track.tofNSigmaEl()) < diffCuts.maxnSigmaTOF()) {
167  return true;
168  }
169  if (TMath::Abs(track.tofNSigmaMu()) < diffCuts.maxnSigmaTOF()) {
170  return true;
171  }
172  if (TMath::Abs(track.tofNSigmaPi()) < diffCuts.maxnSigmaTOF()) {
173  return true;
174  }
175  if (TMath::Abs(track.tofNSigmaKa()) < diffCuts.maxnSigmaTOF()) {
176  return true;
177  }
178  if (TMath::Abs(track.tofNSigmaPr()) < diffCuts.maxnSigmaTOF()) {
179  return true;
180  }
181  }
182  return false;
183 }
