41 struct ProduceMinMaxPt {
42  Produces<aod::MinMaxPt> minmaxpt;
43  float minpt;
44  float maxpt;
45 
46  void process(aod::Collision const& collision, aod::Tracks const& tracks)
47  {
48  // reset minpt / maxpt
49  minpt = 1.E3;
50  maxpt = 0.;
51 
52  // loop over tracks of the collision
53  LOGF(info, "Collision %d number of tracks %d", collision.index(), tracks.size());
54  for (auto& track : tracks) {
55  if (track.pt() < minpt) {
56  minpt = track.pt();
57  }
58  if (track.pt() > maxpt) {
59  maxpt = track.pt();
60  }
61  }
62  LOGF(info, " ptmin %f ptmax %f", minpt, maxpt);
63 
64  // update table minmax
65  minmaxpt(minpt, maxpt);
66  }
67 };
