52  bool isSelected(T1 observable)
53  {
54  switch (mSelType) {
55  case (femtoDreamSelection::SelectionType::kUpperLimit):
56  return (observable < mSelVal);
57  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
58  return (std::abs(observable) < mSelVal);
59  break;
60  case (femtoDreamSelection::SelectionType::kLowerLimit):
61  return (observable > mSelVal);
62  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
63  return (std::abs(observable) > mSelVal);
64  break;
65  case (femtoDreamSelection::SelectionType::kEqual):
67  return (std::abs(observable - mSelVal) < std::abs(mSelVal * 1e-6));
68  break;
69  }
70  return false;
71  }
