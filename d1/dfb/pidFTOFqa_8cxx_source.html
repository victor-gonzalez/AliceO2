65  template <typename T>
66  void makelogaxis(T h)
67  {
68  if (logAxis == 0) {
69  return;
70  }
71  const int nbins = h->GetNbinsX();
72  double binp[nbins + 1];
73  double max = h->GetXaxis()->GetBinUpEdge(nbins);
74  double min = h->GetXaxis()->GetBinLowEdge(1);
75  if (min <= 0) {
76  min = 0.00001;
77  }
78  double lmin = TMath::Log10(min);
79  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
80  for (int i = 0; i < nbins; i++) {
81  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
82  }
83  binp[nbins] = max + 1;
84  h->GetXaxis()->Set(nbins, binp);
85  }

126  using Trks = soa::Join<aod::Tracks, aod::FTOFTracksIndex, aod::TracksExtra>;
127  void process(const Trks& tracks,
128  const aod::McTrackLabels& labels,
129  const aod::FTOFs&,
130  const aod::McParticles& mcParticles,
131  const aod::Collisions& colls)
132  {
133  for (const auto& col : colls) {
134  histos.fill(HIST("event/vertexz"), col.posZ());
135  }
136  for (const auto& track : tracks) {
137  histos.fill(HIST("event/length"), track.ftof().ftofLength());
138  histos.fill(HIST("event/eta"), track.eta());
139  if (!track.has_ftof()) {
140  continue;
141  }
142  if (track.ftof().ftofLength() < minLength) {
143  continue;
144  }
145  if (track.ftof().ftofLength() > maxLength) {
146  continue;
147  }
148  if (track.eta() > maxEta || track.eta() < minEta) {
149  continue;
150  }
151  const auto mcParticle = labels.iteratorAt(track.globalIndex()).mcParticle();
152  if (pdgCode != 0 && abs(mcParticle.pdgCode()) != pdgCode) {
153  continue;
154  }
155  if (useOnlyPhysicsPrimary == 1 && !MC::isPhysicalPrimary(mcParticles, mcParticle)) { // Selecting primaries
156  histos.fill(HIST("p/Sec"), track.p());
157  continue;
158  }
159  histos.fill(HIST("p/Prim"), track.p());
160  histos.fill(HIST("p/Unselected"), track.p());
161  histos.fill(HIST("pt/Unselected"), track.pt());
162  histos.fill(HIST("qa/length"), track.ftof().ftofLength());
163  histos.fill(HIST("qa/signal"), track.ftof().ftofSignal());
164  histos.fill(HIST("qa/signalvsP"), track.p(), track.ftof().ftofSignal());
165  histos.fill(HIST("qa/deltaEl"), track.p(), track.ftof().ftofDeltaEl());
166  histos.fill(HIST("qa/deltaMu"), track.p(), track.ftof().ftofDeltaMu());
167  histos.fill(HIST("qa/deltaPi"), track.p(), track.ftof().ftofDeltaPi());
168  histos.fill(HIST("qa/deltaKa"), track.p(), track.ftof().ftofDeltaKa());
169  histos.fill(HIST("qa/deltaPr"), track.p(), track.ftof().ftofDeltaPr());
170  histos.fill(HIST("qa/nsigmaEl"), track.p(), track.ftof().ftofNsigmaEl());
171  histos.fill(HIST("qa/nsigmaMu"), track.p(), track.ftof().ftofNsigmaMu());
172  histos.fill(HIST("qa/nsigmaPi"), track.p(), track.ftof().ftofNsigmaPi());
173  histos.fill(HIST("qa/nsigmaKa"), track.p(), track.ftof().ftofNsigmaKa());
174  histos.fill(HIST("qa/nsigmaPr"), track.p(), track.ftof().ftofNsigmaPr());
175  }
