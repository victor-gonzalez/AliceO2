164 {
165  switch (pid) {

169  case o2::track::PID::Muon:
170  return track.tpcNSigmaMu();
171  break;
172  case o2::track::PID::Pion:
173  return track.tpcNSigmaPi();
174  break;
175  case o2::track::PID::Kaon:
176  return track.tpcNSigmaKa();
177  break;
178  case o2::track::PID::Proton:
179  return track.tpcNSigmaPr();
180  break;
181  case o2::track::PID::Deuteron:

224 template <typename T>
225 bool FemtoDreamTrackSelection::isSelectedMinimal(T const& track)
226 {
227  const auto pT = track.pt();
228  const auto eta = track.eta();
229  const auto tpcNClsF = track.tpcNClsFound();
230  const auto tpcRClsC = track.tpcCrossedRowsOverFindableCls();
231  const auto tpcNClsC = track.tpcNClsCrossedRows();
232  const auto tpcNClsS = track.tpcNClsShared();
233  const auto dcaXY = track.dcaXY();
234  const auto dcaZ = track.dcaZ();
235 
237 
238  const static int nPtMinSel = getNSelections(femtoDreamTrackSelection::kpTMin);
239  const static int nPtMaxSel = getNSelections(femtoDreamTrackSelection::kpTMax);
240  const static int nEtaSel = getNSelections(femtoDreamTrackSelection::kEtaMax);
241  const static int nTPCnMinSel = getNSelections(femtoDreamTrackSelection::kTPCnClsMin);
242  const static int nTPCfMinSel = getNSelections(femtoDreamTrackSelection::kTPCfClsMin);
243  const static int nTPCcMinSel = getNSelections(femtoDreamTrackSelection::kTPCcRowsMin);
244  const static int nTPCsMaxSel = getNSelections(femtoDreamTrackSelection::kTPCsClsMax);
245  const static int nDCAxyMaxSel = getNSelections(femtoDreamTrackSelection::kDCAxyMax);
246  const static int nDCAzMaxSel = getNSelections(femtoDreamTrackSelection::kDCAzMax);
247 
248  const static float pTMin = getMinimalSelection(femtoDreamTrackSelection::kpTMin, femtoDreamSelection::kLowerLimit);
249  const static float pTMax = getMinimalSelection(femtoDreamTrackSelection::kpTMax, femtoDreamSelection::kUpperLimit);
250  const static float etaMax = getMinimalSelection(femtoDreamTrackSelection::kEtaMax, femtoDreamSelection::kAbsUpperLimit);
251  const static float nClsMin = getMinimalSelection(femtoDreamTrackSelection::kTPCnClsMin, femtoDreamSelection::kLowerLimit);
252  const static float fClsMin = getMinimalSelection(femtoDreamTrackSelection::kTPCfClsMin, femtoDreamSelection::kLowerLimit);
253  const static float cTPCMin = getMinimalSelection(femtoDreamTrackSelection::kTPCcRowsMin, femtoDreamSelection::kLowerLimit);
254  const static float sTPCMax = getMinimalSelection(femtoDreamTrackSelection::kTPCsClsMax, femtoDreamSelection::kUpperLimit);
255  const static float dcaXYMax = getMinimalSelection(femtoDreamTrackSelection::kDCAxyMax, femtoDreamSelection::kAbsUpperLimit);
256  const static float dcaZMax = getMinimalSelection(femtoDreamTrackSelection::kDCAzMax, femtoDreamSelection::kAbsUpperLimit);
257 
258  if (nPtMinSel > 0 && pT < pTMin) {
259  return false;
260  }
261  if (nPtMaxSel > 0 && pT > pTMax) {
262  return false;
263  }
264  if (nEtaSel > 0 && std::abs(eta) > etaMax) {
265  return false;
266  }
267  if (nTPCnMinSel > 0 && tpcNClsF < nClsMin) {
268  return false;
269  }
270  if (nTPCfMinSel > 0 && tpcRClsC < fClsMin) {
271  return false;
272  }
273  if (nTPCcMinSel > 0 && tpcNClsC < cTPCMin) {
274  return false;
275  }
276  if (nTPCsMaxSel > 0 && tpcNClsS > sTPCMax) {
277  return false;
278  }
279  if (nDCAxyMaxSel > 0 && std::abs(dcaXY) > dcaXYMax) {
280  return false;
281  }
282  if (nDCAzMaxSel > 0 && std::abs(dcaZ) > dcaZMax) {
283  return false;
284  }
285  return true;
286 }

288 template <typename T>
289 uint64_t FemtoDreamTrackSelection::getCutContainer(T const& track)
290 {
291  uint64_t output = 0;
292  size_t counter = 2; // first two slots reserved for track/v0/cascade encoding
293  const auto sign = track.sign();
294  const auto pT = track.pt();
295  const auto eta = track.eta();
296  const auto tpcNClsF = track.tpcNClsFound();
297  const auto tpcRClsC = track.tpcCrossedRowsOverFindableCls();
298  const auto tpcNClsC = track.tpcNClsCrossedRows();
299  const auto tpcNClsS = track.tpcNClsShared();
300  const auto dcaXY = track.dcaXY();
301  const auto dcaZ = track.dcaZ();
302  std::vector<float> pidTPC, pidTOF;
303  for (auto it : mPIDspecies) {
304  pidTPC.push_back(getNsigmaTPC(track, it));
305  pidTOF.push_back(getNsigmaTOF(track, it));
306  }
307 
308  float observable;
309  for (auto& sel : mSelections) {
310  const auto selVariable = sel.getSelectionVariable();
311  if (selVariable == femtoDreamTrackSelection::kPIDnSigmaMax) {
313  for (size_t i = 0; i < pidTPC.size(); ++i) {
314  auto pidTPCVal = pidTPC.at(i);
315  auto pidTOFVal = pidTOF.at(i);
316  sel.checkSelectionSetBit(pidTPCVal, output, counter);
317  auto pidComb = std::sqrt(pidTPCVal * pidTPCVal + pidTOFVal * pidTOFVal);
318  sel.checkSelectionSetBit(pidComb, output, counter);
319  }
320 
321  } else {
323  switch (selVariable) {
324  case (femtoDreamTrackSelection::kSign):
325  observable = sign;
326  break;
327  case (femtoDreamTrackSelection::kpTMin):
328  case (femtoDreamTrackSelection::kpTMax):
329  observable = pT;
330  break;
331  case (femtoDreamTrackSelection::kEtaMax):
332  observable = eta;
333  break;
334  case (femtoDreamTrackSelection::kTPCnClsMin):
335  observable = tpcNClsF;
336  break;
337  case (femtoDreamTrackSelection::kTPCfClsMin):
338  observable = tpcRClsC;
339  break;
340  case (femtoDreamTrackSelection::kTPCcRowsMin):
341  observable = tpcNClsC;
342  break;
343  case (femtoDreamTrackSelection::kTPCsClsMax):
344  observable = tpcNClsS;
345  break;
346  case (femtoDreamTrackSelection::kDCAxyMax):
347  observable = dcaXY;
348  break;
349  case (femtoDreamTrackSelection::kDCAzMax):
350  observable = dcaZ;
351  break;
352  case (femtoDreamTrackSelection::kPIDnSigmaMax):
353  break;
354  }
355  sel.checkSelectionSetBit(observable, output, counter);
356  }
357  }
358  return output;
359 }

379 template <typename T>
380 void FemtoDreamTrackSelection::fillCutQA(T const& track, uint64_t cutContainer)
381 {
382  if (mHistogramRegistry) {
383  size_t counter = 2; // first two slots reserved for track/v0/cascade encoding
384  const auto sign = track.sign();
385  const auto pT = track.pt();
386  const auto eta = track.eta();
387  const auto tpcNClsF = track.tpcNClsFound();
388  const auto tpcRClsC = track.tpcCrossedRowsOverFindableCls();
389  const auto tpcNClsC = track.tpcNClsCrossedRows();
390  const auto tpcNClsS = track.tpcNClsShared();
391  const auto dcaXY = track.dcaXY();
392  const auto dcaZ = track.dcaZ();
393  std::vector<float> pidTPC, pidTOF;
394  for (auto it : mPIDspecies) {
395  pidTPC.push_back(getNsigmaTPC(track, it));
396  pidTOF.push_back(getNsigmaTOF(track, it));
397  }
398 
399  femtoDreamTrackSelection::TrackSel oldTrackSel = femtoDreamTrackSelection::kSign;
400  size_t currentTrackSelCounter = 0;
401  size_t pidCounter = 0;
402 
403  for (auto& sel : mSelections) {
404  const auto selVariable = sel.getSelectionVariable();
405  if (oldTrackSel != selVariable) {
406  currentTrackSelCounter = 0;
407  }
408  if (selVariable == femtoDreamTrackSelection::kPIDnSigmaMax) {
410  for (size_t i = 0; i < pidTPC.size(); ++i) {
411  bool isTrueTPC = (cutContainer >> counter) & 1;
412  ++counter;
413  bool isTrueComb = (cutContainer >> counter) & 1;
414  ++counter;
415  auto pidTPCVal = pidTPC.at(i);
416  auto pidTOFVal = pidTOF.at(i);
417  auto pidComb = std::sqrt(pidTPCVal * pidTPCVal + pidTOFVal * pidTOFVal);
418  if (isTrueTPC) {
419  mHistogramRegistry->fill(HIST("TrackCutsQA/pidTPCnsigmaMax"), pidCounter, pidTPCVal);
420  }
421  if (isTrueComb) {
422  mHistogramRegistry->fill(HIST("TrackCutsQA/pidCombnsigmaMax"), pidCounter, pidComb);
423  }
424  ++pidCounter;
425  }
426  ++currentTrackSelCounter;
427  } else {
429  bool isTrue = (cutContainer >> counter) & 1;
430  switch (selVariable) {
431  case (femtoDreamTrackSelection::kSign):
432  if (isTrue) {
433  mHistogramRegistry->fill(HIST("TrackCutsQA/Charge"), currentTrackSelCounter, sign);
434  }
435  break;
436  case (femtoDreamTrackSelection::kpTMin):
437  if (isTrue) {
438  mHistogramRegistry->fill(HIST("TrackCutsQA/pTMin"), currentTrackSelCounter, pT);
439  }
440  break;
441  case (femtoDreamTrackSelection::kpTMax):
442  if (isTrue) {
443  mHistogramRegistry->fill(HIST("TrackCutsQA/pTMax"), currentTrackSelCounter, pT);
444  }
445  break;
446  case (femtoDreamTrackSelection::kEtaMax):
447  if (isTrue) {
448  mHistogramRegistry->fill(HIST("TrackCutsQA/etaMax"), currentTrackSelCounter, eta);
449  }
450  break;
451  case (femtoDreamTrackSelection::kTPCnClsMin):
452  if (isTrue) {
453  mHistogramRegistry->fill(HIST("TrackCutsQA/tpcnClsMin"), currentTrackSelCounter, tpcNClsF);
454  }
455  break;
456  case (femtoDreamTrackSelection::kTPCfClsMin):
457  if (isTrue) {
458  mHistogramRegistry->fill(HIST("TrackCutsQA/tpcfClsMin"), currentTrackSelCounter, tpcRClsC);
459  }
460  break;
461  case (femtoDreamTrackSelection::kTPCcRowsMin):
462  if (isTrue) {
463  mHistogramRegistry->fill(HIST("TrackCutsQA/tpcnRowsMin"), currentTrackSelCounter, tpcNClsC);
464  }
465  break;
466  case (femtoDreamTrackSelection::kTPCsClsMax):
467  if (isTrue) {
468  mHistogramRegistry->fill(HIST("TrackCutsQA/tpcnSharedMax"), currentTrackSelCounter, tpcNClsS);
469  }
470  break;
471  case (femtoDreamTrackSelection::kDCAxyMax):
472  if (isTrue) {
473  mHistogramRegistry->fill(HIST("TrackCutsQA/dcaXYMax"), currentTrackSelCounter, dcaXY);
474  }
475  break;
476  case (femtoDreamTrackSelection::kDCAzMax):
477  if (isTrue) {
478  mHistogramRegistry->fill(HIST("TrackCutsQA/dcaZMax"), currentTrackSelCounter, dcaZ);
479  }
480  break;
481  case (femtoDreamTrackSelection::kPIDnSigmaMax):
482  break;
483  }
484  ++currentTrackSelCounter;
485  ++counter;
486  oldTrackSel = selVariable;
487  }
488  }
489  }
