91  float alphav0(const array<float, 3>& ppos, const array<float, 3>& pneg)
92  {
93  std::array<float, 3> pv0 = {ppos[0] + pneg[0], ppos[1] + pneg[1], ppos[2] + pneg[2]};
94  float momTot = RecoDecay::P(pv0);
95  float lQlNeg = RecoDecay::dotProd(pneg, pv0) / momTot;
96  float lQlPos = RecoDecay::dotProd(ppos, pv0) / momTot;
97  return (lQlPos - lQlNeg) / (lQlPos + lQlNeg); //longitudinal momentum asymmetry
98  }
99 
100  float qtarmv0(const array<float, 3>& ppos, const array<float, 3>& pneg)
101  {
102  std::array<float, 3> pv0 = {ppos[0] + pneg[0], ppos[1] + pneg[1], ppos[2] + pneg[2]};
103  float momTot2 = RecoDecay::P2(pv0);
104  float dp = RecoDecay::dotProd(pneg, pv0);
105  return std::sqrt(RecoDecay::P2(pneg) - dp * dp / momTot2); //qtarm
106  }

108  float phivv0(const array<float, 3>& ppos, const array<float, 3>& pneg, const int cpos, const int cneg, const float bz)
109  {
110  //momentum of e+ and e- in (ax,ay,az) axis. Note that az=0 by definition.
111  //vector product of pep X pem
112  float vpx = 0, vpy = 0, vpz = 0;
113  if (cpos * cneg > 0.) { // Like Sign
114  if (bz * cpos < 0) {
115  vpx = ppos[1] * pneg[2] - ppos[2] * pneg[1];
116  vpy = ppos[2] * pneg[0] - ppos[0] * pneg[2];
117  vpz = ppos[0] * pneg[1] - ppos[2] * pneg[0];
118  } else {
119  vpx = pneg[1] * ppos[2] - pneg[2] * ppos[1];
120  vpy = pneg[2] * ppos[0] - pneg[0] * ppos[2];
121  vpz = pneg[0] * ppos[1] - pneg[2] * ppos[0];
122  }
123  } else { // Unlike Sign
124  if (bz * cpos > 0) {
125  vpx = ppos[1] * pneg[2] - ppos[2] * pneg[1];
126  vpy = ppos[2] * pneg[0] - ppos[0] * pneg[2];
127  vpz = ppos[0] * pneg[1] - ppos[2] * pneg[0];
128  } else {
129  vpx = pneg[1] * ppos[2] - pneg[2] * ppos[1];
130  vpy = pneg[2] * ppos[0] - pneg[0] * ppos[2];
131  vpz = pneg[0] * ppos[1] - pneg[2] * ppos[0];
132  }
133  }
134 
135  //unit vector of pep X pem
136  float vx = vpx / RecoDecay::P(array{vpx, vpy, vpz});
137  float vy = vpy / RecoDecay::P(array{vpx, vpy, vpz});
138  float vz = vpz / RecoDecay::P(array{vpx, vpy, vpz});
139 
140  float px = ppos[0] + pneg[0];
141  float py = ppos[1] + pneg[1];
142  float pz = ppos[2] + pneg[2];
143 
144  //unit vector of (pep+pem)
145  float ux = px / RecoDecay::P(array{px, py, pz});
146  float uy = py / RecoDecay::P(array{px, py, pz});
147  float uz = pz / RecoDecay::P(array{px, py, pz});
148  float ax = uy / RecoDecay::sqrtSumOfSquares(ux, uy);
149  float ay = -ux / RecoDecay::sqrtSumOfSquares(ux, uy);
150 
151  //The third axis defined by vector product (ux,uy,uz)X(vx,vy,vz)
152  float wx = uy * vz - uz * vy;
153  float wy = uz * vx - ux * vz;
154  // by construction, (wx,wy,wz) must be a unit vector. Measure angle between (wx,wy,wz) and (ax,ay,0).
155  // The angle between them should be small if the pair is conversion. This function then returns values close to pi!
156  return TMath::ACos(wx * ax + wy * ay); //phiv in [0,pi] //cosPhiV = wx * ax + wy * ay;
157  }
158 
159  float psipairv0(const array<float, 3>& ppos, const array<float, 3>& pneg, const float bz)
160  {
161  //Following idea to use opening of colinear pairs in magnetic field from e.g. PHENIX to ID conversions.
162  float deltat = TMath::ATan(pneg[2] / (TMath::Sqrt(pneg[0] * pneg[0] + pneg[1] * pneg[1]))) - TMath::ATan(ppos[2] / (TMath::Sqrt(ppos[0] * ppos[0] + ppos[1] * ppos[1]))); //difference of angles of the two daughter tracks with z-axis
163  float pEle = RecoDecay::P(pneg); //absolute momentum val
164  float pPos = RecoDecay::P(ppos); //absolute momentum val
165  float chipair = TMath::ACos(RecoDecay::dotProd(ppos, pneg) / (pEle * pPos)); //Angle between daughter tracks
166  return TMath::Abs(TMath::ASin(deltat / chipair)); //psipair in [0,pi/2]
167  }

169  int processV0(const array<float, 3>& ppos, const array<float, 3>& pneg)
170  {
171  float alpha = alphav0(ppos, pneg);
172  float qt = qtarmv0(ppos, pneg);
173 
174  // Gamma cuts
175  const float cutAlphaG = 0.4;
176  const float cutQTG = 0.03;
177  const float cutAlphaG2[2] = {0.4, 0.8};
178  const float cutQTG2 = 0.02;
179 
180  // K0S cuts
181  const float cutQTK0S[2] = {0.1075, 0.215};
182  const float cutAPK0S[2] = {0.199, 0.8}; // parameters for curved QT cut
183 
184  // Lambda & A-Lambda cuts
185  const float cutQTL = 0.03;
186  const float cutAlphaL[2] = {0.35, 0.7};
187  const float cutAlphaAL[2] = {-0.7, -0.35};
188  const float cutAPL[3] = {0.107, -0.69, 0.5}; // parameters fir curved QT cut
189 
190  // Check for Gamma candidates
191  if (qt < cutQTG) {
192  if ((TMath::Abs(alpha) < cutAlphaG)) {
193  return kGamma;
194  }
195  }
196  if (qt < cutQTG2) {
197  // additional region - should help high pT gammas
198  if ((TMath::Abs(alpha) > cutAlphaG2[0]) && (TMath::Abs(alpha) < cutAlphaG2[1])) {
199  return kGamma;
200  }
201  }
202 
203  // Check for K0S candidates
204  float q = cutAPK0S[0] * TMath::Sqrt(TMath::Abs(1 - alpha * alpha / (cutAPK0S[1] * cutAPK0S[1])));
205  if ((qt > cutQTK0S[0]) && (qt < cutQTK0S[1]) && (qt > q)) {
206  return kK0S;
207  }
208 
209  // Check for Lambda candidates
210  q = cutAPL[0] * TMath::Sqrt(TMath::Abs(1 - ((alpha + cutAPL[1]) * (alpha + cutAPL[1])) / (cutAPL[2] * cutAPL[2])));
211  if ((alpha > cutAlphaL[0]) && (alpha < cutAlphaL[1]) && (qt > cutQTL) && (qt < q)) {
212  return kLambda;
213  }
214 
215  // Check for A-Lambda candidates
216  q = cutAPL[0] * TMath::Sqrt(TMath::Abs(1 - ((alpha - cutAPL[1]) * (alpha - cutAPL[1])) / (cutAPL[2] * cutAPL[2])));
217  if ((alpha > cutAlphaAL[0]) && (alpha < cutAlphaAL[1]) && (qt > cutQTL) && (qt < q)) {
218  return kAntiLambda;
219  }
220 
221  return kUndef;
222  }

270  void process(aod::Collision const& collision, aod::V0s const& V0s, aod::Cascades const& Cascades, FullTracksExt const& tracks)
271  {
272  registry.fill(HIST("hEventCounter"), 0.5);
273  std::array<float, 3> pVtx = {collision.posX(), collision.posY(), collision.posZ()};
274 
275  //Define o2 fitter, 2-prong
276  o2::vertexing::DCAFitterN<2> fitter;
277  fitter.setBz(d_bz);
278  fitter.setPropagateToPCA(true);
279  fitter.setMaxR(200.);
280  fitter.setMinParamChange(1e-3);
281  fitter.setMinRelChi2Change(0.9);
282  fitter.setMaxDZIni(1e9);
283  fitter.setMaxChi2(1e9);
284  fitter.setUseAbsDCA(true); // use d_UseAbsDCA once we want to use the weighted DCA
285 
286  std::map<int, uint8_t> pidmap;
287 
288  int Ngamma = 0;
289  for (auto& V0 : V0s) {
290  if (!(V0.posTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
291  continue;
292  }
293  if (!(V0.negTrack_as<FullTracksExt>().trackType() & o2::aod::track::TPCrefit)) {
294  continue;
295  }
296 
297  if (V0.posTrack_as<FullTracksExt>().tpcNClsCrossedRows() < mincrossedrows) {
298  continue;
299  }
300  if (V0.negTrack_as<FullTracksExt>().tpcNClsCrossedRows() < mincrossedrows) {
301  continue;
302  }
303 
304  if (V0.posTrack_as<FullTracksExt>().tpcChi2NCl() > maxchi2tpc) {
305  continue;
306  }
307  if (V0.negTrack_as<FullTracksExt>().tpcChi2NCl() > maxchi2tpc) {
308  continue;
309  }
310 
311  if (fabs(V0.posTrack_as<FullTracksExt>().dcaXY()) < dcamin) {
312  continue;
313  }
314  if (fabs(V0.negTrack_as<FullTracksExt>().dcaXY()) < dcamin) {
315  continue;
316  }
317 
318  if (fabs(V0.posTrack_as<FullTracksExt>().dcaXY()) > dcamax) {
319  continue;
320  }
321  if (fabs(V0.negTrack_as<FullTracksExt>().dcaXY()) > dcamax) {
322  continue;
323  }
324 
325  if (V0.posTrack_as<FullTracksExt>().sign() * V0.negTrack_as<FullTracksExt>().sign() > 0) { //reject same sign pair
326  continue;
327  }
328 
329  if (V0.posTrack_as<FullTracksExt>().collisionId() != V0.negTrack_as<FullTracksExt>().collisionId()) {
330  continue;
331  }
332 
333  //printf("bcId = %d , collisionId = %d\n",collision.bcId(), V0.negTrack_as<FullTracksExt>().collisionId());
334 
335  registry.fill(HIST("hV0Candidate"), 0.5);
336 
337  std::array<float, 3> pos = {0.};
338  std::array<float, 3> pvec0 = {0.};
339  std::array<float, 3> pvec1 = {0.};
340 
341  int cpos = V0.posTrack_as<FullTracksExt>().sign();
342  int cneg = V0.negTrack_as<FullTracksExt>().sign();
343 
344  auto pTrack = getTrackParCov(V0.posTrack_as<FullTracksExt>());
345  auto nTrack = getTrackParCov(V0.negTrack_as<FullTracksExt>());
346 
347  if (cpos < 0) { //swap charge
348  nTrack = getTrackParCov(V0.posTrack_as<FullTracksExt>());
349  pTrack = getTrackParCov(V0.negTrack_as<FullTracksExt>());
350  }
351 
352  int nCand = fitter.process(pTrack, nTrack);
353  if (nCand != 0) {
354  fitter.propagateTracksToVertex();
355  const auto& vtx = fitter.getPCACandidate();
356  for (int i = 0; i < 3; i++) {
357  pos[i] = vtx[i];
358  }
359  fitter.getTrack(0).getPxPyPzGlo(pvec0); //positive
360  fitter.getTrack(1).getPxPyPzGlo(pvec1); //negative
361  } else {
362  continue;
363  }
364 
365  auto px = pvec0[0] + pvec1[0];
366  auto py = pvec0[1] + pvec1[1];
367  auto pz = pvec0[2] + pvec1[2];
368  auto pt = RecoDecay::sqrtSumOfSquares(pvec0[0] + pvec1[0], pvec0[1] + pvec1[1]);
369  auto eta = RecoDecay::Eta(array{px, py, pz});
370  auto phi = RecoDecay::Phi(px, py);
371 
372  auto V0dca = fitter.getChi2AtPCACandidate(); //distance between 2 legs.
373  auto V0CosinePA = RecoDecay::CPA(pVtx, array{pos[0], pos[1], pos[2]}, array{px, py, pz});
374  auto V0radius = RecoDecay::sqrtSumOfSquares(pos[0], pos[1]);
375 
376  registry.fill(HIST("hV0Pt"), pt);
377  registry.fill(HIST("hV0EtaPhi"), phi, eta);
378  registry.fill(HIST("hDCAxyPosToPV"), V0.posTrack_as<FullTracksExt>().dcaXY());
379  registry.fill(HIST("hDCAxyNegToPV"), V0.negTrack_as<FullTracksExt>().dcaXY());
380 
381  registry.fill(HIST("hV0Radius"), V0radius);
382  registry.fill(HIST("hV0CosPA"), V0CosinePA);
383  registry.fill(HIST("hDCAV0Dau"), V0dca);
384 
385  if (V0dca > dcav0dau) {
386  continue;
387  }
388 
389  if (V0CosinePA < v0cospa) {
390  continue;
391  }
392 
393  if (V0radius < v0Rmin || v0Rmax < V0radius) {
394  continue;
395  }
396 
397  float alpha = alphav0(pvec0, pvec1);
398  float qtarm = qtarmv0(pvec0, pvec1);
399  float phiv = phivv0(pvec0, pvec1, cpos, cneg, d_bz);
400  float psipair = psipairv0(pvec0, pvec1, d_bz);
401 
402  registry.fill(HIST("hV0APplot"), alpha, qtarm);
403  registry.fill(HIST("hV0PhiV"), phiv);
404  registry.fill(HIST("hV0Psi"), psipair);
405 
406  float mGamma = RecoDecay::M(array{pvec0, pvec1}, array{RecoDecay::getMassPDG(kElectron), RecoDecay::getMassPDG(kElectron)});
407  float mK0S = RecoDecay::M(array{pvec0, pvec1}, array{RecoDecay::getMassPDG(kPiPlus), RecoDecay::getMassPDG(kPiPlus)});
408  float mLambda = RecoDecay::M(array{pvec0, pvec1}, array{RecoDecay::getMassPDG(kProton), RecoDecay::getMassPDG(kPiPlus)});
409  float mAntiLambda = RecoDecay::M(array{pvec0, pvec1}, array{RecoDecay::getMassPDG(kPiPlus), RecoDecay::getMassPDG(kProton)});
410 
411  registry.fill(HIST("hMassGamma"), mGamma);
412  registry.fill(HIST("hMassK0S"), mK0S);
413  registry.fill(HIST("hMassLambda"), mLambda);
414  registry.fill(HIST("hMassAntiLambda"), mAntiLambda);
415 
416  int v0id = processV0(pvec0, pvec1);
417  if (v0id < 0) {
418  continue;
419  }
420 
421  if (v0id == kGamma) { //photon conversion
422  if ((70 < V0.posTrack_as<FullTracksExt>().tpcSignal() && V0.posTrack_as<FullTracksExt>().tpcSignal() < 90) && (70 < V0.negTrack_as<FullTracksExt>().tpcSignal() && V0.negTrack_as<FullTracksExt>().tpcSignal() < 90)) {
423  v0Gamma(V0.negTrack_as<FullTracksExt>().collisionId(), pt, eta, phi, mGamma);
424  Ngamma++;
425  }
426  }
427 
428  if (v0id == kGamma && mGamma < 0.01 && TMath::Abs(V0.posTrack_as<FullTracksExt>().tpcNSigmaEl()) < 5 && TMath::Abs(V0.negTrack_as<FullTracksExt>().tpcNSigmaEl()) < 5) { //photon conversion

430  pidmap[V0.negTrackId()] |= (uint8_t(1) << kGamma);
