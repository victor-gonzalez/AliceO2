135  void process(aod::DDbarPairFull const& pairEntries)
136  {
137  for (auto& pairEntry : pairEntries) {
138  //define variables for widely used quantities
139  double deltaPhi = pairEntry.deltaPhi();
140  double deltaEta = pairEntry.deltaEta();
141  double ptD = pairEntry.ptD();
142  double ptDbar = pairEntry.ptDbar();
143  double massD = pairEntry.mD();
144  double massDbar = pairEntry.mDbar();
145 
146  //fill 2D invariant mass plots
147  registry.fill(HIST("hMass2DCorrelationPairs"), massD, massDbar, ptD, ptDbar);
148 
149  //reject entries outside pT ranges of interest
150  int pTBinD = o2::analysis::findBin(binsCorrelations, ptD);
151  int pTBinDbar = o2::analysis::findBin(binsCorrelations, ptDbar);
152  if (pTBinD == -1 || pTBinDbar == -1) { //at least one particle outside accepted pT range
153  continue;
154  }
155 
156  //check if correlation entry belongs to signal region, sidebands or is outside both, and fill correlation plots
157  if (massD > signalRegionInner->at(pTBinD) && massD < signalRegionOuter->at(pTBinD) && massDbar > signalRegionInner->at(pTBinDbar) && massDbar < signalRegionOuter->at(pTBinDbar)) {
158  //in signal region
159  registry.fill(HIST("hCorrel2DVsPtSignalRegion"), deltaPhi, deltaEta, ptD, ptDbar);
160  registry.fill(HIST("hCorrel2DPtIntSignalRegion"), deltaPhi, deltaEta);
161  registry.fill(HIST("hDeltaEtaPtIntSignalRegion"), deltaEta);
162  registry.fill(HIST("hDeltaPhiPtIntSignalRegion"), deltaPhi);
163  registry.fill(HIST("hDeltaPtDDbarSignalRegion"), ptDbar - ptD);
164  registry.fill(HIST("hDeltaPtMaxMinSignalRegion"), std::abs(ptDbar - ptD));
165  }
166 
167  if ((massD > sidebandLeftInner->at(pTBinD) && massD < sidebandLeftOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar)) ||
168  (massD > sidebandRightInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar)) ||
169  (massD > sidebandLeftInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandLeftOuter->at(pTBinDbar)) ||
170  (massD > sidebandLeftInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandRightInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar))) {
171  //in sideband region
172  registry.fill(HIST("hCorrel2DVsPtSidebands"), deltaPhi, deltaEta, ptD, ptDbar);
173  registry.fill(HIST("hCorrel2DPtIntSidebands"), deltaPhi, deltaEta);
174  registry.fill(HIST("hDeltaEtaPtIntSidebands"), deltaEta);
175  registry.fill(HIST("hDeltaPhiPtIntSidebands"), deltaPhi);
176  registry.fill(HIST("hDeltaPtDDbarSidebands"), ptDbar - ptD);
177  registry.fill(HIST("hDeltaPtMaxMinSidebands"), std::abs(ptDbar - ptD));
178  }
179  } //end loop
180  }
181 };

275  void process(aod::DDbarPairFull const& pairEntries)
276  {
277  for (auto& pairEntry : pairEntries) {
278  //define variables for widely used quantities
279  double deltaPhi = pairEntry.deltaPhi();
280  double deltaEta = pairEntry.deltaEta();
281  double ptD = pairEntry.ptD();
282  double ptDbar = pairEntry.ptDbar();
283  double massD = pairEntry.mD();
284  double massDbar = pairEntry.mDbar();
285 
286  //reject entries outside pT ranges of interest
287  int pTBinD = o2::analysis::findBin(binsCorrelations, ptD);
288  int pTBinDbar = o2::analysis::findBin(binsCorrelations, ptDbar);
289  if (pTBinD == -1 || pTBinDbar == -1) { //at least one particle outside accepted pT range
290  continue;
291  }
292 
293  //fill 2D invariant mass plots
294  switch (pairEntry.signalStatus()) {
295  case 0: //D Bkg, Dbar Bkg
296  registry.fill(HIST("hMass2DCorrelationPairsMCRecBkgBkg"), massD, massDbar, ptD, ptDbar);
297  break;
298  case 1: //D Bkg, Dbar Ref
299  registry.fill(HIST("hMass2DCorrelationPairsMCRecBkgRef"), massD, massDbar, ptD, ptDbar);
300  break;
301  case 2: //D Bkg, Dbar Sig
302  registry.fill(HIST("hMass2DCorrelationPairsMCRecBkgSig"), massD, massDbar, ptD, ptDbar);
303  break;
304  case 3: //D Ref, Dbar Bkg
305  registry.fill(HIST("hMass2DCorrelationPairsMCRecRefBkg"), massD, massDbar, ptD, ptDbar);
306  break;
307  case 4: //D Ref, Dbar Ref
308  registry.fill(HIST("hMass2DCorrelationPairsMCRecRefRef"), massD, massDbar, ptD, ptDbar);
309  break;
310  case 5: //D Ref, Dbar Sig
311  registry.fill(HIST("hMass2DCorrelationPairsMCRecRefSig"), massD, massDbar, ptD, ptDbar);
312  break;
313  case 6: //D Sig, Dbar Bkg
314  registry.fill(HIST("hMass2DCorrelationPairsMCRecSigBkg"), massD, massDbar, ptD, ptDbar);
315  break;
316  case 7: //D Sig, Dbar Ref
317  registry.fill(HIST("hMass2DCorrelationPairsMCRecSigRef"), massD, massDbar, ptD, ptDbar);
318  break;
319  case 8: //D Sig, Dbar Sig
320  registry.fill(HIST("hMass2DCorrelationPairsMCRecSigSig"), massD, massDbar, ptD, ptDbar);
321  break;
322  default: //should not happen for MC reco
323  break;
324  }
325 
326  //check if correlation entry belongs to signal region, sidebands or is outside both, and fill correlation plots
327  if (massD > signalRegionInner->at(pTBinD) && massD < signalRegionOuter->at(pTBinD) && massDbar > signalRegionInner->at(pTBinDbar) && massDbar < signalRegionOuter->at(pTBinDbar)) {

329  registry.fill(HIST("hCorrel2DPtIntSignalRegionMCRec"), deltaPhi, deltaEta);
330  registry.fill(HIST("hDeltaEtaPtIntSignalRegionMCRec"), deltaEta);
331  registry.fill(HIST("hDeltaPhiPtIntSignalRegionMCRec"), deltaPhi);
332  registry.fill(HIST("hDeltaPtDDbarSignalRegionMCRec"), ptDbar - ptD);
333  registry.fill(HIST("hDeltaPtMaxMinSignalRegionMCRec"), std::abs(ptDbar - ptD));
334  switch (pairEntry.signalStatus()) {
335  case 0: //D Bkg, Dbar Bkg
336  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecBkgBkg"), deltaPhi, deltaEta, ptD, ptDbar);
337  break;
338  case 1: //D Bkg, Dbar Ref
339  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecBkgRef"), deltaPhi, deltaEta, ptD, ptDbar);
340  break;
341  case 2: //D Bkg, Dbar Sig
342  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecBkgSig"), deltaPhi, deltaEta, ptD, ptDbar);
343  break;
344  case 3: //D Ref, Dbar Bkg
345  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecRefBkg"), deltaPhi, deltaEta, ptD, ptDbar);
346  break;
347  case 4: //D Ref, Dbar Ref
348  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecRefRef"), deltaPhi, deltaEta, ptD, ptDbar);
349  break;
350  case 5: //D Ref, Dbar Sig
351  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecRefSig"), deltaPhi, deltaEta, ptD, ptDbar);
352  break;
353  case 6: //D Sig, Dbar Bkg
354  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecSigBkg"), deltaPhi, deltaEta, ptD, ptDbar);
355  break;
356  case 7: //D Sig, Dbar Ref
357  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecSigRef"), deltaPhi, deltaEta, ptD, ptDbar);
358  break;
359  case 8: //D Sig, Dbar Sig
360  registry.fill(HIST("hCorrel2DVsPtSignalRegionMCRecSigSig"), deltaPhi, deltaEta, ptD, ptDbar);
361  break;
362  default: //should not happen for MC reco
363  break;
364  }
365  }
366 
367  if ((massD > sidebandLeftInner->at(pTBinD) && massD < sidebandLeftOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar)) ||
368  (massD > sidebandRightInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar)) ||
369  (massD > sidebandLeftInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandLeftInner->at(pTBinDbar) && massDbar < sidebandLeftOuter->at(pTBinDbar)) ||
370  (massD > sidebandLeftInner->at(pTBinD) && massD < sidebandRightOuter->at(pTBinD) && massDbar > sidebandRightInner->at(pTBinDbar) && massDbar < sidebandRightOuter->at(pTBinDbar))) {
371  //in sideband region
372  registry.fill(HIST("hCorrel2DPtIntSidebandsMCRec"), deltaPhi, deltaEta);
373  registry.fill(HIST("hDeltaEtaPtIntSidebandsMCRec"), deltaEta);
374  registry.fill(HIST("hDeltaPhiPtIntSidebandsMCRec"), deltaPhi);
375  registry.fill(HIST("hDeltaPtDDbarSidebandsMCRec"), ptDbar - ptD);
376  registry.fill(HIST("hDeltaPtMaxMinSidebandsMCRec"), std::abs(ptDbar - ptD));
377  switch (pairEntry.signalStatus()) {
378  case 0: //D Bkg, Dbar Bkg
379  registry.fill(HIST("hCorrel2DVsPtSidebandsMCRecBkgBkg"), deltaPhi, deltaEta, ptD, ptDbar);
