97  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>::iterator& collision,
98  const o2::aod::McCollisions&,
99  const o2::aod::Tracks& tracks)
100  {
101  if (collision.numContrib() < numberOfContributorsMin) {
102  return;
103  }
104  int nTracks = 0;
105  for (const auto& track : tracks) {
106  if (track.eta() < etaMin || track.eta() > etaMax) {
107  continue;
108  }
109  nTracks++;
110  }
111  histograms.fill(HIST("eventCount"), 0);
112 
113  histograms.fill(HIST("collision/X"), collision.posX());
114  histograms.fill(HIST("collision/Y"), collision.posY());
115  histograms.fill(HIST("collision/Z"), collision.posZ());
116 
117  histograms.fill(HIST("collision/XvsNContrib"), collision.posX(), collision.numContrib());
118  histograms.fill(HIST("collision/YvsNContrib"), collision.posY(), collision.numContrib());
119  histograms.fill(HIST("collision/ZvsNContrib"), collision.posZ(), collision.numContrib());
120 
121  histograms.fill(HIST("collision/numberOfContributors"), collision.numContrib());
122  histograms.fill(HIST("collision/numberOfContributorsVsMult"), collision.numContrib(), nTracks);
123  histograms.fill(HIST("collision/vertexChi2"), collision.chi2());
124 
125  histograms.fill(HIST("covariance/xx"), collision.covXX());
126  histograms.fill(HIST("covariance/xy"), collision.covXY());
127  histograms.fill(HIST("covariance/xz"), collision.covXZ());
128  histograms.fill(HIST("covariance/yy"), collision.covYY());
129  histograms.fill(HIST("covariance/yz"), collision.covYZ());
130  histograms.fill(HIST("covariance/zz"), collision.covZZ());
131 
132  histograms.fill(HIST("multiplicity/numberOfTracks"), nTracks);
133 
134  const auto mcColl = collision.mcCollision();
135  histograms.fill(HIST("resolution/X"), collision.posX() - mcColl.posX(), collision.numContrib());
136  histograms.fill(HIST("resolution/Y"), collision.posY() - mcColl.posY(), collision.numContrib());
137  histograms.fill(HIST("resolution/Z"), collision.posZ() - mcColl.posZ(), collision.numContrib());
138  }
139 };

198  void process(const o2::soa::Join<o2::aod::Tracks, o2::aod::TracksCov, o2::aod::McTrackLabels>& tracks,
199  const o2::aod::McParticles& mcParticles)
200  {
201  for (const auto& t : tracks) {
202  const auto particle = t.mcParticle();
203  if (pdgCodeSel != 0 && particle.pdgCode() != pdgCodeSel) { // Checking PDG code
204  continue;
205  }
206  if (MC::isPhysicalPrimary(mcParticles, particle)) {
207  histos.fill(HIST("trackingPrm/pt"), t.pt());
208  histos.fill(HIST("trackingPrm/eta"), t.eta());
209  histos.fill(HIST("trackingPrm/phi"), t.phi());
210  } else {
211  histos.fill(HIST("trackingSec/pt"), t.pt());
212  histos.fill(HIST("trackingSec/eta"), t.eta());
213  histos.fill(HIST("trackingSec/phi"), t.phi());
214  }
215  histos.fill(HIST("tracking/pt"), t.pt());
216  histos.fill(HIST("tracking/eta"), t.eta());
217  histos.fill(HIST("tracking/phi"), t.phi());
218  }
219  for (const auto& particle : mcParticles) {
220  if (pdgCodeSel != 0 && particle.pdgCode() != pdgCodeSel) { // Checking PDG code
221  continue;
222  }
223  histos.fill(HIST("particle/pt"), particle.pt());
224  histos.fill(HIST("particle/eta"), particle.eta());
225  histos.fill(HIST("particle/phi"), particle.phi());
226  if (MC::isPhysicalPrimary(mcParticles, particle)) {
227  histos.fill(HIST("particlePrm/pt"), particle.pt());
228  histos.fill(HIST("particlePrm/eta"), particle.eta());
229  histos.fill(HIST("particlePrm/phi"), particle.phi());
230  } else {
231  histos.fill(HIST("particleSec/pt"), particle.pt());
232  histos.fill(HIST("particleSec/eta"), particle.eta());
233  histos.fill(HIST("particleSec/phi"), particle.phi());
234  }
235  }
236  }

359  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>::iterator& collision,
360  const o2::soa::Join<o2::aod::Tracks, o2::aod::TracksCov, o2::aod::McTrackLabels>& tracks,
361  const o2::aod::McParticles& mcParticles, const o2::aod::McCollisions& mcCollisions)
362  {
363  const VertexBase primaryVertex = getPrimaryVertex(collision);
364  DCA dca;
365  // FIXME: get this from CCDB
366  constexpr float magneticField{5.0}; // in kG
367  constexpr float toMicrometers = 10000.f; // Conversion from [cm] to [mum]
368  float impactParameterRPhi = -999.f;
369  float impactParameterRPhiError = -999.f;
370  float impactParameterZ = -999.f;
371  float impactParameterErrorZ = -999.f;
372 
373  for (const auto& track : tracks) {
374 
375  const auto particle = track.mcParticle();
376  if (pdgCodeSel != 0 && particle.pdgCode() != pdgCodeSel) {
377  continue;
378  }
379  if (useOnlyPhysicsPrimary && !MC::isPhysicalPrimary(mcParticles, particle)) {
380  continue;
381  }
382  const double deltaPt = track.pt() - particle.pt();
383  histos.fill(HIST("pt/ptDiffRecGen"), deltaPt);
384 
385  const double deltaPtOverPt = deltaPt / particle.pt();
386 
387  histos.fill(HIST("pt/ptResolution"), deltaPtOverPt);
388  histos.fill(HIST("pt/ptResolutionVsPt"), track.pt(), deltaPtOverPt);
389  histos.fill(HIST("pt/ptResolutionVsEta"), track.eta(), deltaPtOverPt);
390  histos.fill(HIST("pt/ptResolutionVsPhi"), track.phi(), deltaPtOverPt);
391 
392  const double deltaEta = track.eta() - particle.eta();
393  histos.fill(HIST("eta/etaDiffRecGen"), deltaEta);
394  histos.fill(HIST("eta/etaDiffRecGenVsEtaGen"), deltaEta, particle.eta());
395  histos.fill(HIST("eta/etaDiffRecGenVsEtaRec"), deltaEta, track.eta());
396 
397  histos.fill(HIST("phi/phiDiffRecGen"), track.phi() - particle.phi());
398 
399  if (getTrackParCov(track).propagateToDCA(primaryVertex, magneticField, &dca, 100.)) { // Check that the propagation is successfull
400  impactParameterRPhi = toMicrometers * dca.getY();
