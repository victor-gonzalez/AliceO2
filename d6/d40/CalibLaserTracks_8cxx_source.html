39 void CalibLaserTracks::processTrack(const TrackTPC& track)
40 {
41  if (track.hasBothSidesClusters()) {
42  return;
43  }
44 
45  auto parOutAtLtr = track.getOuterParam();
46 
47  // track should have been alreay propagated close to the laser mirrors
48  if (parOutAtLtr.getX() < 220) {
49  return;
50  }
51 
52  // recalculate z position based on trigger or CE position
53  float zTrack = parOutAtLtr.getZ();
54 
55  // TODO: calculation has to be improved
56  if (mTriggerPos < 0) {
57  // use CE for time 0
58  const float zOffset = (track.getTime0() + mTriggerPos) * mZbinWidth * mDriftV + 250;
59  //printf("time0: %.2f, trigger pos: %d, zTrack: %.2f, zOffset: %.2f\n", track.getTime0(), mTriggerPos, zTrack, zOffset);
60  zTrack += zOffset;
61  parOutAtLtr.setZ(zTrack);
62  } else if (mTriggerPos > 0) {
63  }
64 
65  if (std::abs(zTrack) > 300) {
66  return;
67  }
68 
69  // try association with ideal laser track and rotate parameters
70  const int side = track.hasCSideClusters();
71  const int laserTrackID = findLaserTrackID(parOutAtLtr, side);
72  if (laserTrackID < 0 || laserTrackID >= LaserTrack::NumberOfTracks) {
73  return;
74  }
75 
76  LaserTrack ltr = mLaserTracks.getTrack(laserTrackID);
77  parOutAtLtr.rotateParam(ltr.getAlpha());
78  parOutAtLtr.propagateParamTo(ltr.getX(), mBz);
79 
80  mZmatchPairsTF.emplace_back(TimePair{ltr.getZ(), parOutAtLtr.getZ(), mTFtime});
81 
82  if (mWriteDebugTree) {
83  if (!mDebugStream) {
84  mDebugStream = std::make_unique<o2::utils::TreeStreamRedirector>("CalibLaserTracks_debug.root", "recreate");
85  }
86 
87  *mDebugStream << "ltrMatch"
88  << "ltr=" << ltr // matched ideal laser track
89  << "trOutLtr=" << parOutAtLtr // track rotated and propagated to ideal track position
90  << "\n";
91  }
92 }

95 int CalibLaserTracks::findLaserTrackID(TrackPar outerParam, int side)
96 {
97  //const auto phisec = getPhiNearbySectorEdge(outerParam);
98  const auto phisec = getPhiNearbyLaserRod(outerParam, side);
99  if (!outerParam.rotateParam(phisec)) {
100  return -1;
101  }
102 
103  if (side < 0) {
104  side = outerParam.getZ() < 0;
105  }
106  const int rod = std::nearbyint((phisec - LaserTrack::FirstRodPhi[side]) / LaserTrack::RodDistancePhi);
107  //printf("\n\nside: %i\n", side);
108  const auto xyzGlo = outerParam.getXYZGlo();
109  auto phi = std::atan2(xyzGlo.Y(), xyzGlo.X());
110  o2::math_utils::bringTo02Pi(phi);
111  //printf("rod: %d (phisec: %.2f, phi: %.2f\n", rod, phisec, phi);
112  int bundle = -1;
113  int beam = -1;
114  float mindist = 1000;
115 
116  const auto outerParamZ = std::abs(outerParam.getZ());
117  //printf("outerParamZ: %.2f\n", outerParamZ);
118  for (size_t i = 0; i < LaserTrack::CoarseBundleZPos.size(); ++i) {
119  const float dist = std::abs(outerParamZ - LaserTrack::CoarseBundleZPos[i]);
120  //printf("laserZ: %.2f (%.2f, %.2f)\n", LaserTrack::CoarseBundleZPos[i], dist, mindist);
121  if (dist < mindist) {
122  mindist = dist;
123  bundle = i;
124  }
125  }
126 
127  if (bundle < 0) {
128  return -1;
129  }
130 
131  //printf("bundle: %i\n", bundle);
132 
133  const auto outerParamsInBundle = mLaserTracks.getTracksInBundle(side, rod, bundle);
134  mindist = 1000;
135  for (int i = 0; i < outerParamsInBundle.size(); ++i) {
136  const auto louterParam = outerParamsInBundle[i];
137  if (i == 0) {
138  outerParam.propagateParamTo(louterParam.getX(), mBz);
139  }
140  const float dist = std::abs(outerParam.getSnp() - louterParam.getSnp());
141  if (dist < mindist) {
142  mindist = dist;
143  beam = i;
144  }
145  }
146 
147  //printf("beam: %i (%.4f)\n", beam, mindist);
148  if (mindist > 0.01) {
149  return -1;
150  }
151 
152  const int trackID = LaserTrack::NumberOfTracks / 2 * side +
153  LaserTrack::BundlesPerRod * LaserTrack::TracksPerBundle * rod +
154  LaserTrack::TracksPerBundle * bundle +
155  beam;
156 
157  //printf("trackID: %d\n", trackID);
158 
159  return trackID;
160 }

163 float CalibLaserTracks::getPhiNearbySectorEdge(const TrackPar& param)
164 {
165  // rotate to nearest laser bundle
166  const auto xyzGlo = param.getXYZGlo();
167  auto phi = std::atan2(xyzGlo.Y(), xyzGlo.X());
168  o2::math_utils::bringTo02PiGen(phi);
169  const auto phisec = std::nearbyint(phi / LaserTrack::SectorSpanRad) * LaserTrack::SectorSpanRad;
170  //printf("%.2f : %.2f\n", phi, phisec);
171  return (phisec);
172 }

174 //______________________________________________________________________________
175 float CalibLaserTracks::getPhiNearbyLaserRod(const TrackPar& param, int side)
176 {
177  const auto xyzGlo = param.getXYZGlo();
178  auto phi = std::atan2(xyzGlo.Y(), xyzGlo.X()) - LaserTrack::FirstRodPhi[side % 2];
179  o2::math_utils::bringTo02PiGen(phi);
180  phi = std::nearbyint(phi / LaserTrack::RodDistancePhi) * LaserTrack::RodDistancePhi + LaserTrack::FirstRodPhi[side % 2];
181  o2::math_utils::bringTo02PiGen(phi);
182  //printf("%.2f : %.2f\n", phi, phisec);
183  return phi;
184 }

249 TimePair CalibLaserTracks::fit(const std::vector<TimePair>& trackMatches) const
250 {
251  if (!trackMatches.size()) {
252  return TimePair();
253  }
254 
255  static TLinearFitter fit(2, "pol1");
256  fit.StoreData(false);
257  fit.ClearPoints();
258 
259  float meanTime = 0;
260  for (const auto& point : trackMatches) {
261  double x = point.x1;
262  double y = point.x2;
263  fit.AddPoint(&x, y);
264 
265  meanTime += point.time;
266  }
267 
268  meanTime /= float(trackMatches.size());
269 
270  const float robustFraction = 0.9;
271  const int minPoints = 6;
272 
273  if (trackMatches.size() < size_t(minPoints / robustFraction)) {
274  return TimePair({0, 0, meanTime});
275  }
276 
277  //fit.EvalRobust(robustFraction);
278  fit.Eval();
279 
280  TimePair retVal;
281  retVal.x1 = float(fit.GetParameter(0));
282  retVal.x2 = float(fit.GetParameter(1));
283  retVal.time = meanTime;
284 
285  return retVal;
286 }
