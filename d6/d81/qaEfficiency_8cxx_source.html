46 template <typename T>
47 void makelogaxis(T h)
48 {
49  const int nbins = h->GetNbinsX();
50  double binp[nbins + 1];
51  double max = h->GetXaxis()->GetBinUpEdge(nbins);
52  double min = h->GetXaxis()->GetBinLowEdge(1);
53  if (min <= 0) {
54  min = 0.01;
55  }
56  double lmin = TMath::Log10(min);
57  double ldelta = (TMath::Log10(max) - lmin) / ((double)nbins);
58  for (int i = 0; i < nbins; i++) {
59  binp[i] = TMath::Exp(TMath::Log(10) * (lmin + i * ldelta));
60  }
61  binp[nbins] = max + 1;
62  h->GetXaxis()->Set(nbins, binp);
63 }

188  void process(const o2::soa::Join<o2::aod::Collisions, o2::aod::McCollisionLabels>& collisions,
189  const o2::soa::Join<o2::aod::Tracks, o2::aod::McTrackLabels>& tracks,
190  const o2::aod::McCollisions& mcCollisions,
191  const o2::aod::McParticles& mcParticles)
192  {
193 
194  auto rejectParticle = [&](auto p, auto h, const int& selBinOffset = 0) {
195  if ((p.pt() < ptMin || p.pt() > ptMax)) { // Check pt
196  return true;
197  }
198  histos.fill(h, 2 + selBinOffset);
199  if ((p.eta() < etaMin || p.eta() > etaMax)) { // Check eta
200  return true;
201  }
202  histos.fill(h, 3 + selBinOffset);
203  if ((p.phi() < phiMin || p.phi() > phiMax)) { // Check phi
204  return true;
205  }
206  histos.fill(h, 4 + selBinOffset);
207  if ((selPrim == 1) && (!MC::isPhysicalPrimary(mcParticles, p))) { // Requiring is physical primary
208  return true;
209  }
210  histos.fill(h, 5 + selBinOffset);
211 
212  // Selecting PDG code
213  switch (pdgSign) {
214  case 0:
215  if (abs(p.pdgCode()) != pdg) {
216  return true;
217  }
218  break;
219  case 1:
220  if (p.pdgCode() != pdg) {
221  return true;
222  }
223  break;
224  case -1:
225  if (p.pdgCode() != -pdg) {
226  return true;
227  }
228  break;
229  default:
230  LOG(FATAL) << "Provide pdgSign as 0, 1, -1. Provided: " << pdgSign.value;
231  break;
232  }
233  histos.fill(h, 6 + selBinOffset);
234 
235  return false;
236  };
237 
238  std::vector<int64_t> recoEvt(collisions.size());
239  int nevts = 0;
240  for (const auto& collision : collisions) {
241  histos.fill(HIST("eventSelection"), 1);
242  if (collision.numContrib() < nMinNumberOfContributors) {
243  continue;
244  }
245  histos.fill(HIST("eventSelection"), 2);
246  const auto mcCollision = collision.mcCollision();
247  if ((mcCollision.posZ() < vertexZMin || mcCollision.posZ() > vertexZMax)) {
248  continue;
249  }
250  histos.fill(HIST("eventSelection"), 3);
251  recoEvt[nevts++] = mcCollision.globalIndex();
252  }
253  recoEvt.resize(nevts);
254 
255  std::vector<int64_t> recoTracks(tracks.size());
256  int ntrks = 0;
257  for (const auto& track : tracks) {
258  histos.fill(HIST("trackSelection"), 1);
259  const auto mcParticle = track.mcParticle();
260 
261  if (rejectParticle(mcParticle, HIST("trackSelection"))) {
262  continue;
263  }
264 
265  if (noFakes) { // Selecting tracks with no fake hits
266  bool hasFake = false;
267  for (int i = 0; i < 10; i++) { // From ITS to TPC
268  if (track.mcMask() & 1 << i) {
269  hasFake = true;
270  break;
271  }
272  }
273  if (hasFake) {
274  continue;
275  }
276  }
277  histos.fill(HIST("trackSelection"), 7);
278  histos.fill(HIST("pt/num"), mcParticle.pt());
279  histos.fill(HIST("eta/num"), mcParticle.eta());
280  histos.fill(HIST("phi/num"), mcParticle.phi());
281  recoTracks[ntrks++] = mcParticle.globalIndex();
282  }
283 
284  for (const auto& mcParticle : mcParticles) {
285  histos.fill(HIST("partSelection"), 1);
286  const auto evtReconstructed = std::find(recoEvt.begin(), recoEvt.end(), mcParticle.mcCollision().globalIndex()) != recoEvt.end();
287  if (!evtReconstructed) {
288  continue;
289  }
290  histos.fill(HIST("partSelection"), 2);
291  if (rejectParticle(mcParticle, HIST("partSelection"), 1)) {
292  continue;
293  }
294 
295  if (makeEff) {
296  const auto particleReconstructed = std::find(recoTracks.begin(), recoTracks.end(), mcParticle.globalIndex()) != recoTracks.end();
297  static_cast<TEfficiency*>(list->At(0))->Fill(particleReconstructed, mcParticle.pt());
298  static_cast<TEfficiency*>(list->At(1))->Fill(particleReconstructed, mcParticle.p());
299  static_cast<TEfficiency*>(list->At(2))->Fill(particleReconstructed, mcParticle.eta());
300  static_cast<TEfficiency*>(list->At(3))->Fill(particleReconstructed, mcParticle.phi());
301  static_cast<TEfficiency*>(list->At(4))->Fill(particleReconstructed, mcParticle.pt(), mcParticle.eta());
302  }
303  histos.fill(HIST("pt/den"), mcParticle.pt());
304  histos.fill(HIST("eta/den"), mcParticle.eta());
305  histos.fill(HIST("phi/den"), mcParticle.phi());
306  }
307  }
