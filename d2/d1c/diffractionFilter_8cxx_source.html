54  // Productions

56

58  MutableConfigurable<cutHolder> diffCuts{"cfgDiffCuts", {}, "Diffractive events cut object"};
59 
60  void init(o2::framework::InitContext&)
61  {
62  diffCuts->SetisRun2(true);
63  }
64 
65  // DG selector
66  DGSelector dgSelector;
67 
68  // some general Collisions and Tracks filter
69  Filter collisionFilter = nabs(aod::collision::posZ) < diffCuts->maxPosz();
70 
71  using BCs = soa::Join<aod::BCs, aod::Run2BCInfos, aod::BcSels, aod::Run2MatchedToBCSparse>;
72  using BC = BCs::iterator;
73  using CCs = soa::Filtered<soa::Join<aod::Collisions, aod::EvSels>>;
74  using CC = CCs::iterator;
75  using TCs = soa::Join<aod::Tracks, aod::TracksExtra, aod::pidTPCEl, aod::pidTPCMu, aod::pidTPCPi, aod::pidTPCKa, aod::pidTPCPr, aod::pidTOFEl, aod::pidTOFMu, aod::pidTOFPi, aod::pidTOFKa, aod::pidTOFPr, aod::TrackSelection>;
76  using MUs = aod::FwdTracks;
77 
78  void process(CC const& collision,
79  BCs& bcs,
80  TCs& tracks,
81  MUs& muons,
82  aod::Zdcs& zdcs,
83  aod::FT0s& ft0s,
84  aod::FV0As& fv0as,
85  aod::FV0Cs& fv0cs,
86  aod::FDDs& fdds)
87  {
88  // nominal BC
89  auto bc = collision.bc_as<BCs>();
90 
91  // Range of BCs relevant for past-future protection
92  // bcNominal +- deltaBC
93  auto nBCMin = bc.globalBC() - diffCuts->deltaBC();
94  auto nBCMax = bc.globalBC() + diffCuts->deltaBC();
95  Partition<BCs> bcRange = aod::bc::globalBC >= nBCMin && aod::bc::globalBC <= nBCMax;
96  bcRange.bindTable(bcs);
97  bcRange.bindExternalIndices(&zdcs, &ft0s, &fv0as, &fv0cs, &fdds);
98 
99  // fill filterTable
100  auto isDGEvent = dgSelector.IsSelected(diffCuts, collision, bc, bcRange, tracks, muons);
101  filterTable(isDGEvent);
102  }
103 };
