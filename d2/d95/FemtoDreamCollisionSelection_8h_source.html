95 template <typename T>
96 bool FemtoDreamCollisionSelection::isSelected(T const& col)
97 {
98  if (std::abs(col.posZ()) > mZvtxMax) {
99  return false;
100  }
101 
102  if (mCheckTrigger && col.alias()[mTrigger] != 1) {
103  return false;
104  }
105 
106  if (mCheckOffline && col.sel7() != 1) {
107  return false;
108  }
109 
110  return true;
111 }
