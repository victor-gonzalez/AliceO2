211  template <typename T>
212  bool isSelectedTrack(const T& hfTrack, const array<float, 2>& dca, const int candType)
213  {
214  auto pTBinTrack = findBin(pTBinsTrack, hfTrack.pt());
215  if (pTBinTrack == -1) {
216  return false;
217  }
218 
219  if (std::abs(dca[0]) < cutsSingleTrack[candType].get(pTBinTrack, "min_dcaxytoprimary")) {
220  return false; //minimum DCAxy
221  }
222  if (std::abs(dca[0]) > cutsSingleTrack[candType].get(pTBinTrack, "max_dcaxytoprimary")) {
223  return false; //maximum DCAxy
224  }
225  return true;
226  }

228  void process(aod::Collision const& collision,
229  MY_TYPE1 const& tracks
230 #ifdef MY_DEBUG
231  ,
232  aod::McParticles& mcParticles
233 #endif
234  )
235  {
236  math_utils::Point3D<float> vtxXYZ(collision.posX(), collision.posY(), collision.posZ());
237  for (auto& track : tracks) {
238 
239 #ifdef MY_DEBUG
240  auto indexBach = track.mcParticleId();
241  // LOG(INFO) << "Checking label " << indexBach;
242  bool isProtonFromLc = isProtonFromLcFunc(indexBach, indexProton);
243 
244 #endif
245 
246  MY_DEBUG_MSG(isProtonFromLc, LOG(INFO) << "\nWe found the proton " << indexBach);
247 
248  int statusProng = BIT(CandidateType::NCandidateTypes) - 1; // selection flag , all bits on
249  bool cutStatus[CandidateType::NCandidateTypes][nCuts];
250  if (debug) {
251  for (int iCandType = 0; iCandType < CandidateType::NCandidateTypes; iCandType++) {
252  for (int iCut = 0; iCut < nCuts; iCut++) {
253  cutStatus[iCandType][iCut] = true;
254  }
255  }
256  }
257 
258  auto trackPt = track.pt();
259  auto trackEta = track.eta();
260 
261  if (fillHistograms) {
262  registry.get<TH1>(HIST("hPtNoCuts"))->Fill(trackPt);
263  }
264 
265  int iDebugCut = 2;
266  // pT cut
267  if (trackPt < pTMinTrack2Prong) {
268  CLRBIT(statusProng, CandidateType::Cand2Prong); // set the nth bit to 0
269  if (debug) {
270  cutStatus[CandidateType::Cand2Prong][0] = false;
271  if (fillHistograms) {
272  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand2Prong + iDebugCut);
273  }
274  }
275  }
276  if (trackPt < pTMinTrack3Prong) {
277  CLRBIT(statusProng, CandidateType::Cand3Prong);
278  if (debug) {
279  cutStatus[CandidateType::Cand3Prong][0] = false;
280  if (fillHistograms) {
281  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand3Prong + iDebugCut);
282  }
283  }
284  }
285  MY_DEBUG_MSG(isProtonFromLc, LOG(INFO) << "proton " << indexBach << " pt = " << trackPt << " (cut " << ptMinTrackBach << ")");
286 
287  if (trackPt < ptMinTrackBach) {
288  CLRBIT(statusProng, CandidateType::CandV0bachelor);
289  if (debug) {
290  cutStatus[CandidateType::CandV0bachelor][0] = false;
291  if (fillHistograms) {
292  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::CandV0bachelor + iDebugCut);
293  }
294  }
295  }
296 
297  iDebugCut = 3;
298  // eta cut
299  if ((debug || TESTBIT(statusProng, CandidateType::Cand2Prong)) && std::abs(trackEta) > etaMax2Prong) {
300  CLRBIT(statusProng, CandidateType::Cand2Prong);
301  if (debug) {
302  cutStatus[CandidateType::Cand2Prong][1] = false;
303  if (fillHistograms) {
304  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand2Prong + iDebugCut);
305  }
306  }
307  }
308  if ((debug || TESTBIT(statusProng, CandidateType::Cand3Prong)) && std::abs(trackEta) > etaMax3Prong) {
309  CLRBIT(statusProng, CandidateType::Cand3Prong);
310  if (debug) {
311  cutStatus[CandidateType::Cand3Prong][1] = false;
312  if (fillHistograms) {
313  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand3Prong + iDebugCut);
314  }
315  }
316  }
317  MY_DEBUG_MSG(isProtonFromLc, LOG(INFO) << "proton " << indexBach << " eta = " << trackEta << " (cut " << etaMaxBach << ")");
318 
319  if ((debug || TESTBIT(statusProng, CandidateType::CandV0bachelor)) && std::abs(trackEta) > etaMaxBach) {
320  CLRBIT(statusProng, CandidateType::CandV0bachelor);
321  if (debug) {
322  cutStatus[CandidateType::CandV0bachelor][1] = false;
323  if (fillHistograms) {
324  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::CandV0bachelor + iDebugCut);
325  }
326  }
327  }
328 
329  // quality cut
330  MY_DEBUG_MSG(isProtonFromLc, LOG(INFO) << "proton " << indexBach << " tpcNClsFound = " << track.tpcNClsFound() << " (cut " << tpcNClsFound.value << ")");
331 
332  iDebugCut = 4;
333  if (doCutQuality.value && (debug || statusProng > 0)) { // FIXME to make a more complete selection e.g track.flags() & o2::aod::track::TPCrefit && track.flags() & o2::aod::track::GoldenChi2 &&
334  UChar_t clustermap = track.itsClusterMap();
335  if (!(track.tpcNClsFound() >= tpcNClsFound.value && // is this the number of TPC clusters? It should not be used
336  track.flags() & o2::aod::track::ITSrefit &&
337  (TESTBIT(clustermap, 0) || TESTBIT(clustermap, 1)))) {
338  statusProng = 0;
339  MY_DEBUG_MSG(isProtonFromLc, LOG(INFO) << "proton " << indexBach << " did not pass clusters cut");
340  if (debug) {
341  for (int iCandType = 0; iCandType < CandidateType::NCandidateTypes; iCandType++) {
342  cutStatus[iCandType][2] = false;
343  if (fillHistograms) {
344  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * iCandType + iDebugCut);
345  }
346  }
347  }
348  }
349  }
350 
351  iDebugCut = 5;
352  // DCA cut
353  array<float, 2> dca;
354  if ((debug || statusProng > 0)) {
355  auto trackparvar0 = getTrackParCov(track);
356  if (!trackparvar0.propagateParamToDCA(vtxXYZ, bz, &dca, 100.)) { // get impact parameters
357  statusProng = 0;
358  }
359  if ((debug || TESTBIT(statusProng, CandidateType::Cand2Prong)) && !isSelectedTrack(track, dca, CandidateType::Cand2Prong)) {
360  CLRBIT(statusProng, CandidateType::Cand2Prong);
361  if (debug) {
362  cutStatus[CandidateType::Cand2Prong][3] = false;
363  if (fillHistograms) {
364  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand2Prong + iDebugCut);
365  }
366  }
367  }
368  if ((debug || TESTBIT(statusProng, CandidateType::Cand3Prong)) && !isSelectedTrack(track, dca, CandidateType::Cand3Prong)) {
369  CLRBIT(statusProng, CandidateType::Cand3Prong);
370  if (debug) {
371  cutStatus[CandidateType::Cand3Prong][3] = false;
372  if (fillHistograms) {
373  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand3Prong + iDebugCut);
374  }
375  }
376  }
377  if ((debug || TESTBIT(statusProng, CandidateType::CandV0bachelor)) && !isSelectedTrack(track, dca, CandidateType::CandV0bachelor)) {
378  CLRBIT(statusProng, CandidateType::CandV0bachelor);
379  if (debug) {
380  cutStatus[CandidateType::CandV0bachelor][3] = false;
381  if (fillHistograms) {
382  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::CandV0bachelor + iDebugCut);
383  }
384  }
385  }
386  }
387  MY_DEBUG_MSG(isProtonFromLc, LOG(INFO) << "statusProng = " << statusProng; printf("\n"));
388 
389  // fill histograms
390  if (fillHistograms) {
391  iDebugCut = 1;
392  if (TESTBIT(statusProng, CandidateType::Cand2Prong)) {
393  registry.get<TH1>(HIST("hPtCuts2Prong"))->Fill(trackPt);
394  registry.get<TH1>(HIST("hEtaCuts2Prong"))->Fill(trackEta);
395  registry.get<TH2>(HIST("hDCAToPrimXYVsPtCuts2Prong"))->Fill(trackPt, dca[0]);
396  if (debug) {
397  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand2Prong + iDebugCut);
398  }
399  }
400  if (TESTBIT(statusProng, CandidateType::Cand3Prong)) {
401  registry.get<TH1>(HIST("hPtCuts3Prong"))->Fill(trackPt);
402  registry.get<TH1>(HIST("hEtaCuts3Prong"))->Fill(trackEta);
403  registry.get<TH2>(HIST("hDCAToPrimXYVsPtCuts3Prong"))->Fill(trackPt, dca[0]);
404  if (debug) {
405  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::Cand3Prong + iDebugCut);
406  }
407  }
408  if (TESTBIT(statusProng, CandidateType::CandV0bachelor)) {
409  MY_DEBUG_MSG(isProtonFromLc, LOG(INFO) << "Will be kept: Proton from Lc " << indexBach);
410  registry.get<TH1>(HIST("hPtCutsV0bachelor"))->Fill(trackPt);
411  registry.get<TH1>(HIST("hEtaCutsV0bachelor"))->Fill(trackEta);
412  registry.get<TH2>(HIST("hDCAToPrimXYVsPtCutsV0bachelor"))->Fill(trackPt, dca[0]);
413  if (debug) {
414  registry.get<TH1>(HIST("hRejTracks"))->Fill((nCuts + 1) * CandidateType::CandV0bachelor + iDebugCut);
415  }
416  }
417  }
418 
419  // fill table row
420  rowSelectedTrack(statusProng, dca[0], dca[1]);
421  }
422  }
