160  template <typename TTRACKS, typename TCOLL, typename TV0>
161  bool processPhoton(TCOLL const& theCollision, const TV0& theV0)
162  {
163  fillHistogramsBeforeCuts(theV0);
164 
165  auto lTrackPos = theV0.template posTrack_as<TTRACKS>(); //positive daughter
166  auto lTrackNeg = theV0.template negTrack_as<TTRACKS>(); //negative daughter
167 
168  // apply track cuts
169  if (!(trackPassesCuts(lTrackPos) && trackPassesCuts(lTrackNeg))) {
170  return kFALSE;
171  }
172 
173  float lV0CosinePA = theV0.v0cosPA(theCollision.posX(), theCollision.posY(), theCollision.posZ());
174 
175  // apply photon cuts
176  if (!passesPhotonCuts(theV0, lV0CosinePA)) {
177  return kFALSE;
178  }
179 
180  fillHistogramsAfterCuts(theV0, lTrackPos, lTrackNeg, lV0CosinePA);
181 
182  return kTRUE;
183  }

185  void processData(collisionEvSelIt const& theCollision,
186  aod::V0Datas const& theV0s,
187  tracksAndTPCInfo const& theTracks)
188  {

191  registry.fill(HIST("hEvents"), 1);

193  }
194  registry.fill(HIST("hEvents"), 3);
195 
196  for (auto& lV0 : theV0s) {

198  continue;
199  }
200  }

203  template <typename T>
204  void fillHistogramsBeforeCuts(const T& theV0)
205  {
206  // fill some QA histograms before any cuts
207  registry.fill(HIST("beforeCuts/hPtRec"), theV0.pt());
208  registry.fill(HIST("beforeCuts/hEtaRec"), theV0.eta());
209  registry.fill(HIST("beforeCuts/hPhiRec"), theV0.phi());
210  registry.fill(HIST("beforeCuts/hConvPointR"), theV0.v0radius());
211  registry.fill(HIST("IsPhotonSelected"), kPhotonIn);

214  template <typename T>
215  bool trackPassesCuts(const T& theTrack)
216  {
217 
218  // single track eta cut
219  if (TMath::Abs(theTrack.eta()) > fEtaCut) {
220  registry.fill(HIST("IsPhotonSelected"), kTrackEta);
221  return kFALSE;
222  }
223 
224  // single track pt cut
225  if (theTrack.pt() < fSinglePtCut) {
226  registry.fill(HIST("IsPhotonSelected"), kTrackPt);
227  return kFALSE;
228  }
229 
230  if (!(selectionPIDTPC_track(theTrack))) {
231  return kFALSE;
232  }
233 
234  if (theTrack.tpcFoundOverFindableCls() < fMinTPCFoundOverFindableCls) {
235  registry.fill(HIST("IsPhotonSelected"), kTPCFoundOverFindableCls);
236  return kFALSE;
237  }
238 
239  if (theTrack.tpcCrossedRowsOverFindableCls() < fMinTPCCrossedRowsOverFindableCls) {
240  registry.fill(HIST("IsPhotonSelected"), kTPCCrossedRowsOverFindableCls);
241  return kFALSE;
242  }
243 
244  return kTRUE;
245  }
246 
247  template <typename T>
248  bool passesPhotonCuts(const T& theV0, float theV0CosinePA)
249  {
250  if (theV0.v0radius() < fMinR || theV0.v0radius() > fMaxR) {
251  registry.fill(HIST("IsPhotonSelected"), kV0Radius);
252  return kFALSE;
253  }
254 
255  if (!ArmenterosQtCut(theV0.alpha(), theV0.qtarm(), theV0.pt())) {
256  registry.fill(HIST("IsPhotonSelected"), kArmenteros);
257  return kFALSE;
258  }
259 
260  if (TMath::Abs(theV0.psipair()) > fPsiPairCut) {
261  registry.fill(HIST("IsPhotonSelected"), kPsiPair);
262  return kFALSE;
263  }
264 
265  if (theV0CosinePA < fCosPAngleCut) {
266  registry.fill(HIST("IsPhotonSelected"), kCosinePA);
267  return kFALSE;
268  }
269 
270  return kTRUE;
271  }

273  template <typename TV0, typename TTRACK>
274  void fillHistogramsAfterCuts(const TV0& theV0, const TTRACK& theTrackPos, const TTRACK& theTrackNeg, float theV0CosinePA)
275  {
276  registry.fill(HIST("IsPhotonSelected"), kPhotonOut);
277 
278  registry.fill(HIST("hPtRec"), theV0.pt());
279  registry.fill(HIST("hEtaRec"), theV0.eta());
280  registry.fill(HIST("hPhiRec"), theV0.phi());
281  registry.fill(HIST("hConvPointR"), theV0.v0radius());
282 
283  registry.fill(HIST("hTPCdEdxSigEl"), theTrackNeg.p(), theTrackNeg.tpcNSigmaEl());
284  registry.fill(HIST("hTPCdEdxSigEl"), theTrackPos.p(), theTrackPos.tpcNSigmaEl());
285  registry.fill(HIST("hTPCdEdxSigPi"), theTrackNeg.p(), theTrackNeg.tpcNSigmaPi());
286  registry.fill(HIST("hTPCdEdxSigPi"), theTrackPos.p(), theTrackPos.tpcNSigmaPi());
287 
288  registry.fill(HIST("hTPCdEdx"), theTrackNeg.p(), theTrackNeg.tpcSignal());
289  registry.fill(HIST("hTPCdEdx"), theTrackPos.p(), theTrackPos.tpcSignal());
290 
291  registry.fill(HIST("hTPCFoundOverFindableCls"), theTrackNeg.tpcFoundOverFindableCls());
292  registry.fill(HIST("hTPCFoundOverFindableCls"), theTrackPos.tpcFoundOverFindableCls());
293 
294  registry.fill(HIST("hTPCCrossedRowsOverFindableCls"), theTrackNeg.tpcCrossedRowsOverFindableCls());
295  registry.fill(HIST("hTPCCrossedRowsOverFindableCls"), theTrackPos.tpcCrossedRowsOverFindableCls());
296 
297  registry.fill(HIST("hArmenteros"), theV0.alpha(), theV0.qtarm());
298  registry.fill(HIST("hPsiPtRec"), theV0.psipair(), theV0.pt());
299 
300  registry.fill(HIST("hCosPAngle"), theV0CosinePA);
301  }

303  Bool_t ArmenterosQtCut(Double_t theAlpha, Double_t theQt, Double_t thePt)
304  {
305  // in AliPhysics this is the cut for if fDo2DQt && fDoQtGammaSelection == 2
306  Float_t lQtMaxPtDep = fQtPtMax * thePt;
307  if (lQtMaxPtDep > fQtMax) {
308  lQtMaxPtDep = fQtMax;
309  }
310  if (!(TMath::Power(theAlpha / fMaxPhotonAsymmetry, 2) + TMath::Power(theQt / lQtMaxPtDep, 2) < 1)) {
311  return kFALSE;
312  }
313  return kTRUE;
314  }

316  template <typename T>
317  bool selectionPIDTPC_track(const T& theTrack)
318  {
319  // TPC Electron Line
320  if (theTrack.tpcNSigmaEl() < fPIDnSigmaBelowElectronLine || theTrack.tpcNSigmaEl() > fPIDnSigmaAboveElectronLine) {
321  registry.fill(HIST("IsPhotonSelected"), kElectronPID);
322  return kFALSE;
323  }
324 
325  // TPC Pion Line
326  if (theTrack.p() > fPIDMinPnSigmaAbovePionLine) {
327  // low pt Pion rej
328  if (theTrack.p() < fPIDMaxPnSigmaAbovePionLine) {
329  if (theTrack.tpcNSigmaEl() > fPIDnSigmaBelowElectronLine && theTrack.tpcNSigmaEl() < fPIDnSigmaAboveElectronLine && theTrack.tpcNSigmaPi() < fPIDnSigmaAbovePionLine) {
330  registry.fill(HIST("IsPhotonSelected"), kPionRejLowMom);
331  return kFALSE;
332  }
333  }
334  // High Pt Pion rej
335  else {
336  if (theTrack.tpcNSigmaEl() > fPIDnSigmaBelowElectronLine && theTrack.tpcNSigmaEl() < fPIDnSigmaAboveElectronLine && theTrack.tpcNSigmaPi() < fPIDnSigmaAbovePionLineHighP) {
337  registry.fill(HIST("IsPhotonSelected"), kPionRejHighMom);
338  return kFALSE;
339  }
340  }
341  }
342  return kTRUE;
343  }
