120  void process(aod::FilteredFullCollision const& col,
121  aod::FilteredFullTracks const& tracks)
122  {
123 
124  if (!colCuts.isSelected(col)) {
125  return;
126  }
127  const auto vtxZ = col.posZ();
128  const auto mult = col.multV0M();
129  const auto spher = colCuts.computeSphericity(col, tracks);
130  colCuts.fillQA(col);
131  outputCollision(vtxZ, mult, spher);
132 
133  for (auto& track : tracks) {
134  if (!trackCuts.isSelectedMinimal(track)) {
135  continue;
136  }
137  trackCuts.fillQA(track);
138  auto cutContainer = trackCuts.getCutContainer(track);
139  if (cutContainer > 0) {
140  trackCuts.fillCutQA(track, cutContainer);
141  outputTracks(outputCollision.lastIndex(), track.pt(), track.eta(), track.phi(), cutContainer, track.dcaXY());
142  if (ConfDebugOutput) {
143  outputDebugTracks(outputCollision.lastIndex(),
144  track.sign(), track.tpcNClsFound(),
145  track.tpcNClsFindable(),
146  track.tpcNClsCrossedRows(), track.tpcNClsShared(), track.dcaXY(), track.dcaZ(),
147  track.tpcNSigmaEl(), track.tpcNSigmaPi(), track.tpcNSigmaKa(), track.tpcNSigmaPr(), track.tpcNSigmaDe(),
148  track.tofNSigmaEl(), track.tofNSigmaPi(), track.tofNSigmaKa(), track.tofNSigmaPr(), track.tofNSigmaDe());
149  }
150  }
151  }
152  }
153 };
