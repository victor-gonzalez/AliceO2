50  void fillSelectionHistogram(const char* name)
51  {
52  int nBins = mSelections.size();
53  mHistogramRegistry->add(name, "; Cut; Value", kTH1F, {{nBins, 0, static_cast<double>(nBins)}});
54  auto hist = mHistogramRegistry->get<TH1>(HIST("TrackCuts/cuthist"));

72  void setSelection(std::vector<FemtoDreamSelection<T1, T2>>& sels)
73  {
74  switch (sels.at(0).getSelectionType()) {
75  case (femtoDreamSelection::SelectionType::kUpperLimit):
76  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
77  std::sort(sels.begin(), sels.end(), [](FemtoDreamSelection<T1, T2> a, FemtoDreamSelection<T1, T2> b) {
78  return a.getSelectionValue() > b.getSelectionValue();
79  });
80  break;
81  case (femtoDreamSelection::SelectionType::kLowerLimit):
82  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
83  case (femtoDreamSelection::SelectionType::kEqual):
84  std::sort(sels.begin(), sels.end(), [](FemtoDreamSelection<T1, T2> a, FemtoDreamSelection<T1, T2> b) {
85  return a.getSelectionValue() < b.getSelectionValue();
86  });
87  break;
88  }
89  for (const auto sel : sels) {
90  mSelections.push_back(sel);
91  }
92  }

94  T1 getMinimalSelection(T2 selVar, femtoDreamSelection::SelectionType selType)
95  {
96  T1 minimalSel;
97  switch (selType) {
98  case (femtoDreamSelection::SelectionType::kUpperLimit):
99  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
100  minimalSel = -999.e9;
101  break;
102  case (femtoDreamSelection::SelectionType::kLowerLimit):
103  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
104  case (femtoDreamSelection::SelectionType::kEqual):
105  minimalSel = 999.e9;
106  break;
107  }
108 
109  for (auto sel : mSelections) {
110  if (sel.getSelectionVariable() == selVar) {
111  switch (sel.getSelectionType()) {
112  case (femtoDreamSelection::SelectionType::kUpperLimit):
113  case (femtoDreamSelection::SelectionType::kAbsUpperLimit):
114  if (minimalSel < sel.getSelectionValue()) {
115  minimalSel = sel.getSelectionValue();
116  }
117  break;
118  case (femtoDreamSelection::SelectionType::kLowerLimit):
119  case (femtoDreamSelection::SelectionType::kAbsLowerLimit):
120  case (femtoDreamSelection::SelectionType::kEqual):
121  if (minimalSel > sel.getSelectionValue()) {
122  minimalSel = sel.getSelectionValue();
123  }
124  break;
125  }
126  }
127  }
128  return minimalSel;
129  }
