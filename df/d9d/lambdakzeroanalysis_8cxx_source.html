71  void process(aod::Collision const& collision, aod::V0Datas const& fullV0s)
72  {
73 
74  for (auto& v0 : fullV0s) {
75  registry.fill(HIST("hMassK0Short"), v0.mK0Short());
76  registry.fill(HIST("hMassLambda"), v0.mLambda());
77  registry.fill(HIST("hMassAntiLambda"), v0.mAntiLambda());
78 
79  registry.fill(HIST("hV0Radius"), v0.v0radius());
80  registry.fill(HIST("hV0CosPA"), v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()));
81  registry.fill(HIST("hDCAPosToPV"), v0.dcapostopv());
82  registry.fill(HIST("hDCANegToPV"), v0.dcanegtopv());
83  registry.fill(HIST("hDCAV0Dau"), v0.dcaV0daughters());
84  }
85  }

102  void init(InitContext const&)
103  {
104  AxisSpec dcaAxis = {dcaBinning, "DCA (cm)"};
105  AxisSpec ptAxis = {ptBinning, "#it{p}_{T} (GeV/c)"};
106  AxisSpec massAxisK0Short = {200, 0.450f, 0.550f, "Inv. Mass (GeV)"};
107  AxisSpec massAxisLambda = {200, 1.015f, 1.215f, "Inv. Mass (GeV)"};
108 
109  registry.add("h3dMassK0ShortDca", "h3dMassK0ShortDca", {HistType::kTH3F, {dcaAxis, ptAxis, massAxisK0Short}});
110  registry.add("h3dMassLambdaDca", "h3dMassLambdaDca", {HistType::kTH3F, {dcaAxis, ptAxis, massAxisLambda}});
111  registry.add("h3dMassAntiLambdaDca", "h3dMassAntiLambdaDca", {HistType::kTH3F, {dcaAxis, ptAxis, massAxisLambda}});
112  }
113 
114  //Selection criteria
115  Configurable<double> v0cospa{"v0cospa", 0.995, "V0 CosPA"}; //double -> N.B. dcos(x)/dx = 0 at x=0)
116  Configurable<float> dcav0dau{"dcav0dau", 1.0, "DCA V0 Daughters"};
117  Configurable<float> dcanegtopv{"dcanegtopv", .1, "DCA Neg To PV"};
118  Configurable<float> dcapostopv{"dcapostopv", .1, "DCA Pos To PV"};
119  Configurable<float> v0radius{"v0radius", 5.0, "v0radius"};
120  Configurable<float> rapidity{"rapidity", 0.5, "rapidity"};
121  Configurable<int> saveDcaHist{"saveDcaHist", 0, "saveDcaHist"};
122 
123  Filter preFilterV0 = nabs(aod::v0data::dcapostopv) > dcapostopv&& nabs(aod::v0data::dcanegtopv) > dcanegtopv&& aod::v0data::dcaV0daughters < dcav0dau;

125  void process(soa::Join<aod::Collisions, aod::EvSels, aod::Cents>::iterator const& collision, soa::Filtered<aod::V0Datas> const& fullV0s)
126  {
127  if (!collision.alias()[kINT7]) {
128  return;
129  }
130  if (!collision.sel7()) {
131  return;
132  }
133 
134  for (auto& v0 : fullV0s) {
135  //FIXME: could not find out how to filter cosPA and radius variables (dynamic columns)
136  if (v0.v0radius() > v0radius && v0.v0cosPA(collision.posX(), collision.posY(), collision.posZ()) > v0cospa) {
137  if (TMath::Abs(v0.yLambda()) < rapidity) {
138  registry.fill(HIST("h3dMassLambda"), collision.centV0M(), v0.pt(), v0.mLambda());
139  registry.fill(HIST("h3dMassAntiLambda"), collision.centV0M(), v0.pt(), v0.mAntiLambda());
140  if (saveDcaHist == 1) {
141  registry.fill(HIST("h3dMassLambdaDca"), v0.dcaV0daughters(), v0.pt(), v0.mLambda());
142  registry.fill(HIST("h3dMassAntiLambdaDca"), v0.dcaV0daughters(), v0.pt(), v0.mAntiLambda());
143  }
144  }
145  if (TMath::Abs(v0.yK0Short()) < rapidity) {
146  registry.fill(HIST("h3dMassK0Short"), collision.centV0M(), v0.pt(), v0.mK0Short());
147  if (saveDcaHist == 1) {
148  registry.fill(HIST("h3dMassK0ShortDca"), v0.dcaV0daughters(), v0.pt(), v0.mK0Short());
149  }
150  }
151  }
152  }
153  }
154 };
