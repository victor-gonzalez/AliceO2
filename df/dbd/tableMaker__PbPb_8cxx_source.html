139  void process(MyEvent const& collision, soa::Filtered<MyMuons> const& tracksMuon, aod::BCs const& bcs, soa::Filtered<MyBarrelTracks> const& tracksBarrel)
140  {
141  uint32_t triggerAliases = 0;
142  for (int i = 0; i < kNaliases; i++) {
143  if (collision.alias()[i] > 0) {
144  triggerAliases |= (uint32_t(1) << i);
145  }
146  }
147  uint64_t tag = 0; // TODO: add here available computed event cuts (e.g. run2bcinfo().eventCuts()) or other event wise decisions
148 
149  VarManager::ResetValues(0, VarManager::kNEventWiseVariables, fValues);
150  VarManager::FillEvent<eventFillMap>(collision, fValues); // extract event information and place it in the fValues array
151  fHistMan->FillHistClass("Event_BeforeCuts", fValues); // automatically fill all the histograms in the class Event
152 
153  if (!fEventCut->IsSelected(fValues)) {
154  return;
155  }
156 
157  fHistMan->FillHistClass("Event_AfterCuts", fValues);
158 
159  event(tag, collision.bc().runNumber(), collision.posX(), collision.posY(), collision.posZ(), collision.numContrib());
160  eventExtended(collision.bc().globalBC(), collision.bc().triggerMask(), 0, triggerAliases, fValues[VarManager::kCentVZERO]);
161  eventVtxCov(collision.covXX(), collision.covXY(), collision.covXZ(), collision.covYY(), collision.covYZ(), collision.covZZ(), collision.chi2());
162 
163  uint64_t trackFilteringTag = 0;
164  trackBasic.reserve(tracksBarrel.size());
165  trackBarrel.reserve(tracksBarrel.size());
166  trackBarrelCov.reserve(tracksBarrel.size());
167  trackBarrelPID.reserve(tracksBarrel.size());
168 
169  for (auto& track : tracksBarrel) {
170  VarManager::FillTrack<gkTrackFillMap>(track, fValues);
171  fHistMan->FillHistClass("TrackBarrel_BeforeCuts", fValues);
172  if (!fTrackCut->IsSelected(fValues)) {
173  continue;
174  }
175  fHistMan->FillHistClass("TrackBarrel_AfterCuts", fValues);
176 
177  etaH->Fill(track.eta());
178  if (track.isGlobalTrack()) {
179  trackFilteringTag |= (uint64_t(1) << 0);
180  }
181  if (track.isGlobalTrackSDD()) {
182  trackFilteringTag |= (uint64_t(1) << 1);
183  }
184  trackBasic(event.lastIndex(), track.globalIndex(), trackFilteringTag, track.pt(), track.eta(), track.phi(), track.sign());
185  trackBarrel(track.tpcInnerParam(), track.flags(), track.itsClusterMap(), track.itsChi2NCl(),
186  track.tpcNClsFindable(), track.tpcNClsFindableMinusFound(), track.tpcNClsFindableMinusCrossedRows(),
187  track.tpcNClsShared(), track.tpcChi2NCl(),
188  track.trdChi2(), track.trdPattern(), track.tofChi2(),
189  track.length(), track.dcaXY(), track.dcaZ());
190  trackBarrelCov(track.x(), track.alpha(), track.y(), track.z(), track.snp(), track.tgl(), track.signed1Pt(),
191  track.cYY(), track.cZY(), track.cZZ(), track.cSnpY(), track.cSnpZ(),
192  track.cSnpSnp(), track.cTglY(), track.cTglZ(), track.cTglSnp(), track.cTglTgl(),
193  track.c1PtY(), track.c1PtZ(), track.c1PtSnp(), track.c1PtTgl(), track.c1Pt21Pt2());
194  trackBarrelPID(track.tpcSignal(),
195  track.tpcNSigmaEl(), track.tpcNSigmaMu(),
196  track.tpcNSigmaPi(), track.tpcNSigmaKa(), track.tpcNSigmaPr(),
197  track.beta(),
198  track.tofNSigmaEl(), track.tofNSigmaMu(),
199  track.tofNSigmaPi(), track.tofNSigmaKa(), track.tofNSigmaPr(),
200  track.trdSignal());
201  }
202 
203  // build the muon tables
204  muonBasic.reserve(tracksMuon.size());
205  muonExtra.reserve(tracksMuon.size());
206  muonCov.reserve(tracksMuon.size());
207  for (auto& muon : tracksMuon) {
208  VarManager::FillTrack<gkMuonFillMap>(muon, fValues);
209  fHistMan->FillHistClass("Muons_BeforeCuts", fValues);
210  if (!fMuonCut->IsSelected(fValues)) {
211  continue;
212  }
213  fHistMan->FillHistClass("Muons_AfterCuts", fValues);
214 
215  muonBasic(event.lastIndex(), trackFilteringTag, muon.pt(), muon.eta(), muon.phi(), muon.sign());
216  muonExtra(muon.nClusters(), muon.pDca(), muon.rAtAbsorberEnd(),
217  muon.chi2(), muon.chi2MatchMCHMID(), muon.chi2MatchMCHMFT(),
218  muon.matchScoreMCHMFT(), muon.matchMFTTrackID(), muon.matchMCHTrackID());
219  muonCov(muon.x(), muon.y(), muon.z(), muon.phi(), muon.tgl(), muon.signed1Pt(),
220  muon.cXX(), muon.cXY(), muon.cYY(), muon.cPhiX(), muon.cPhiY(), muon.cPhiPhi(),
221  muon.cTglX(), muon.cTglY(), muon.cTglPhi(), muon.cTglTgl(), muon.c1PtX(), muon.c1PtY(),
222  muon.c1PtPhi(), muon.c1PtTgl(), muon.c1Pt21Pt2());
223  }
224  }
