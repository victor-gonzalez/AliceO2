50 struct HFCandidateCreatorCascade {
51 
52  Produces<aod::HfCandCascBase> rowCandidateBase;
53 
54  Configurable<double> bZ{"bZ", 5., "magnetic field"};
55  Configurable<bool> propDCA{"propDCA", true, "create tracks version propagated to PCA"};
56  Configurable<double> maxR{"maxR", 200., "reject PCA's above this radius"};
57  Configurable<double> maxDZIni{"maxDZIni", 4., "reject (if>0) PCA candidate if tracks DZ exceeds threshold"};
58  Configurable<double> minParamChange{"minParamChange", 1.e-3, "stop iterations if largest change of any X is smaller than this"};
59  Configurable<double> minRelChi2Change{"minRelChi2Change", 0.9, "stop iterations is chi2/chi2old > this"};
60  Configurable<bool> doValPlots{"doValPlots", true, "do validation plots"};
61 
62  // for debugging
63 #ifdef MY_DEBUG
64  Configurable<std::vector<int>> indexK0Spos{"indexK0Spos", {729, 2866, 4754, 5457, 6891, 7824, 9243, 9810}, "indices of K0S positive daughters, for debug"};
65  Configurable<std::vector<int>> indexK0Sneg{"indexK0Sneg", {730, 2867, 4755, 5458, 6892, 7825, 9244, 9811}, "indices of K0S negative daughters, for debug"};
66  Configurable<std::vector<int>> indexProton{"indexProton", {717, 2810, 4393, 5442, 6769, 7793, 9002, 9789}, "indices of protons, for debug"};
67 #endif
68 
69  OutputObj<TH1F> hmass2{TH1F("hmass2", "2-prong candidates;inv. mass (#pi K) (GeV/#it{c}^{2});entries", 500, 0., 5.)};
70  OutputObj<TH1F> hCovPVXX{TH1F("hCovPVXX", "2-prong candidates;XX element of cov. matrix of prim. vtx. position (cm^{2});entries", 100, 0., 1.e-4)};
71  OutputObj<TH1F> hCovSVXX{TH1F("hCovSVXX", "2-prong candidates;XX element of cov. matrix of sec. vtx. position (cm^{2});entries", 100, 0., 0.2)};
72 
73  double massP = RecoDecay::getMassPDG(kProton);
74  double massK0s = RecoDecay::getMassPDG(kK0Short);
75  double massPi = RecoDecay::getMassPDG(kPiPlus);
76  double massLc = RecoDecay::getMassPDG(pdg::Code::kLambdaCPlus);
77  double mass2K0sP{0.};

79  void process(aod::Collisions const& collisions,
80  aod::HfTrackIndexCasc const& rowsTrackIndexCasc,
81  MyBigTracks const& tracks,
82  aod::V0Datas const& V0s
83 #ifdef MY_DEBUG
84  ,
85  aod::McParticles& mcParticles
86 #endif
87  )
88  {
89  // 2-prong vertex fitter
90  o2::vertexing::DCAFitterN<2> df;
91  df.setBz(bZ);
92  df.setPropagateToPCA(propDCA);
93  df.setMaxR(maxR);
94  df.setMaxDZIni(maxDZIni);
95  df.setMinParamChange(minParamChange);
96  df.setMinRelChi2Change(minRelChi2Change);
97  df.setUseAbsDCA(true);
98 
99  // loop over pairs of track indeces
100  for (const auto& casc : rowsTrackIndexCasc) {
101 
102  const auto& bach = casc.index0_as<MyBigTracks>();
103  const auto& v0 = casc.indexV0_as<o2::aod::V0Datas>();
104  const auto& trackV0DaughPos = v0.posTrack_as<MyBigTracks>();
105  const auto& trackV0DaughNeg = v0.negTrack_as<MyBigTracks>();
106 
107 #ifdef MY_DEBUG
108  auto indexBach = bach.mcParticleId();
109  auto indexV0DaughPos = trackV0DaughPos.mcParticleId();
110  auto indexV0DaughNeg = trackV0DaughNeg.mcParticleId();
111  bool isLc = isLcK0SpFunc(indexBach, indexV0DaughPos, indexV0DaughNeg, indexProton, indexK0Spos, indexK0Sneg);
112 #endif
113 
114  MY_DEBUG_MSG(isLc, LOG(INFO) << "Processing the Lc with proton " << indexBach << " trackV0DaughPos " << indexV0DaughPos << " trackV0DaughNeg " << indexV0DaughNeg);
115 
116  auto trackParCovBach = getTrackParCov(bach);
117  auto trackParCovV0DaughPos = getTrackParCov(trackV0DaughPos); // check that MyBigTracks does not need TracksExtended!
118  auto trackParCovV0DaughNeg = getTrackParCov(trackV0DaughNeg); // check that MyBigTracks does not need TracksExtended!
119  trackParCovV0DaughPos.propagateTo(v0.posX(), bZ); // propagate the track to the X closest to the V0 vertex
120  trackParCovV0DaughNeg.propagateTo(v0.negX(), bZ); // propagate the track to the X closest to the V0 vertex
121  const std::array<float, 3> vertexV0 = {v0.x(), v0.y(), v0.z()};
122  const std::array<float, 3> momentumV0 = {v0.px(), v0.py(), v0.pz()};
123  // we build the neutral track to then build the cascade
124  auto trackV0 = o2::dataformats::V0(vertexV0, momentumV0, {0, 0, 0, 0, 0, 0}, trackParCovV0DaughPos, trackParCovV0DaughNeg, {0, 0}, {0, 0}); // build the V0 track (indices for v0 daughters set to 0 for now)
125 
126  auto collision = bach.collision();
127 
128  // reconstruct the cascade secondary vertex
129  if (df.process(trackV0, trackParCovBach) == 0) {
130  MY_DEBUG_MSG(isLc, LOG(INFO) << "Vertexing failed for Lc candidate");
131  // if (isLc) {
132  // LOG(INFO) << "Vertexing failed for Lc with proton " << indexBach << " trackV0DaughPos " << indexV0DaughPos << " trackV0DaughNeg " << indexV0DaughNeg;
133  //}
134  continue;
135  } else {
136  //LOG(INFO) << "Vertexing succeeded for Lc candidate";
137  }
138 
139  const auto& secondaryVertex = df.getPCACandidate();
140  auto chi2PCA = df.getChi2AtPCACandidate();
141  auto covMatrixPCA = df.calcPCACovMatrix().Array();
142  hCovSVXX->Fill(covMatrixPCA[0]); // FIXME: Calculation of errorDecayLength(XY) gives wrong values without this line.
143  // do I have to call "df.propagateTracksToVertex();"?
144  auto trackParVarV0 = df.getTrack(0);
145  auto trackParVarBach = df.getTrack(1);
146 
147  // get track momenta
148  array<float, 3> pVecV0;
149  array<float, 3> pVecBach;
150  trackParVarV0.getPxPyPzGlo(pVecV0);
151  trackParVarBach.getPxPyPzGlo(pVecBach);
152 
153  // get track impact parameters
154  // This modifies track momenta!
155  auto primaryVertex = getPrimaryVertex(collision);
156  auto covMatrixPV = primaryVertex.getCov();
157  hCovPVXX->Fill(covMatrixPV[0]);
158  o2::dataformats::DCA impactParameterV0;
159  o2::dataformats::DCA impactParameterBach;
160  trackParVarV0.propagateToDCA(primaryVertex, bZ, &impactParameterV0); // we do this wrt the primary vtx
161  trackParVarBach.propagateToDCA(primaryVertex, bZ, &impactParameterBach);
162 
163  // get uncertainty of the decay length
164  double phi, theta;
165  getPointDirection(array{collision.posX(), collision.posY(), collision.posZ()}, secondaryVertex, phi, theta);
166  auto errorDecayLength = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, theta) + getRotatedCovMatrixXX(covMatrixPCA, phi, theta));
167  auto errorDecayLengthXY = std::sqrt(getRotatedCovMatrixXX(covMatrixPV, phi, 0.) + getRotatedCovMatrixXX(covMatrixPCA, phi, 0.));
168 
169  // fill candidate table rows
170  MY_DEBUG_MSG(isLc, LOG(INFO) << "IT IS A Lc! Filling for Lc with proton " << indexBach << " trackV0DaughPos " << indexV0DaughPos << " trackV0DaughNeg " << indexV0DaughNeg);
171  rowCandidateBase(collision.globalIndex(),
172  collision.posX(), collision.posY(), collision.posZ(),
173  secondaryVertex[0], secondaryVertex[1], secondaryVertex[2],
174  errorDecayLength, errorDecayLengthXY,
175  chi2PCA,
176  pVecBach[0], pVecBach[1], pVecBach[2],
177  pVecV0[0], pVecV0[1], pVecV0[2],
178  impactParameterBach.getY(), impactParameterV0.getY(),
179  std::sqrt(impactParameterBach.getSigmaY2()), std::sqrt(impactParameterV0.getSigmaY2()),
180  casc.index0Id(), casc.indexV0Id(),
181  casc.hfflag(),
182  v0.x(), v0.y(), v0.z(),
183  //v0.posTrack(), v0.negTrack(), // why this was not fine?
184  trackV0DaughPos.globalIndex(), trackV0DaughNeg.globalIndex(),
185  v0.pxpos(), v0.pypos(), v0.pzpos(),
186  v0.pxneg(), v0.pyneg(), v0.pzneg(),
187  v0.dcaV0daughters(),
188  v0.dcapostopv(),
189  v0.dcanegtopv());
190 
191  // fill histograms
192  if (doValPlots) {
193  // calculate invariant masses
194  mass2K0sP = RecoDecay::M(array{pVecBach, pVecV0}, array{massP, massK0s});
195  hmass2->Fill(mass2K0sP);
196  }
197  }
198  }
199 };

220  void process(aod::HfCandCascade const& candidates,
221  aod::BigTracksMC const& tracks,
222  aod::McParticles const& particlesMC)
223  {
224  int8_t sign = 0;
225  std::vector<int> arrDaughLcIndex;
226  std::array<int, 3> arrDaughLcPDG;
227  std::array<int, 3> arrDaughLcPDGRef = {2212, 211, -211};
228 
229  // Match reconstructed candidates.
230  for (auto& candidate : candidates) {
231 
232  const auto& bach = candidate.index0_as<aod::BigTracksMC>();
233  const auto& trackV0DaughPos = candidate.posTrack_as<aod::BigTracksMC>();
234  const auto& trackV0DaughNeg = candidate.negTrack_as<aod::BigTracksMC>();
235 
236  auto arrayDaughtersV0 = array{trackV0DaughPos, trackV0DaughNeg};
237  auto arrayDaughtersLc = array{bach, trackV0DaughPos, trackV0DaughNeg};
238 
239  // First we check the K0s
240  LOG(DEBUG) << "\n";
241  LOG(DEBUG) << "Checking MC for candidate!";
242  LOG(DEBUG) << "Looking for K0s";
243  auto indexV0DaughPos = trackV0DaughPos.mcParticleId();
244  auto indexV0DaughNeg = trackV0DaughNeg.mcParticleId();
245  auto indexBach = bach.mcParticleId();
246 
247 #ifdef MY_DEBUG
248  bool isLc = isLcK0SpFunc(indexBach, indexV0DaughPos, indexV0DaughNeg, indexProton, indexK0Spos, indexK0Sneg);
249  bool isK0SfromLc = isK0SfromLcFunc(indexV0DaughPos, indexV0DaughNeg, indexK0Spos, indexK0Sneg);
250 #endif
251  MY_DEBUG_MSG(isK0SfromLc, LOG(INFO) << "correct K0S in the Lc daughters: posTrack --> " << indexV0DaughPos << ", negTrack --> " << indexV0DaughNeg);
252 
253  //if (isLc) {
254  RecoDecay::getMatchedMCRec(particlesMC, arrayDaughtersV0, kK0Short, array{+kPiPlus, -kPiPlus}, true, &sign, 1); // does it matter the "acceptAntiParticle" in the K0s case? In principle, there is no anti-K0s
255 
256  if (sign != 0) { // we have already positively checked the K0s
257  // then we check the Lc
258  MY_DEBUG_MSG(sign, LOG(INFO) << "K0S was correct! now we check the Lc");
259  MY_DEBUG_MSG(sign, LOG(INFO) << "index proton = " << indexBach);
260  RecoDecay::getMatchedMCRec(particlesMC, arrayDaughtersLc, pdg::Code::kLambdaCPlus, array{+kProton, +kPiPlus, -kPiPlus}, true, &sign, 3); // 3-levels Lc --> p + K0 --> p + K0s --> p + pi+ pi-
261  MY_DEBUG_MSG(sign, LOG(INFO) << "Lc found with sign " << sign; printf("\n"));
262  }
263 
264  rowMCMatchRec(sign);
265  }
266  //}
267 
268  // Match generated particles.
269  for (auto& particle : particlesMC) {
270  // checking if I have a Lc --> K0S + p
271  RecoDecay::isMatchedMCGen(particlesMC, particle, pdg::Code::kLambdaCPlus, array{+kProton, +kK0Short}, true, &sign, 2);
272  if (sign != 0) {
273  MY_DEBUG_MSG(sign, LOG(INFO) << "Lc in K0S p");
274  arrDaughLcIndex.clear();
275  // checking that the final daughters (decay depth = 3) are p, pi+, pi-
276  RecoDecay::getDaughters(particlesMC, particle, &arrDaughLcIndex, arrDaughLcPDGRef, 3); // best would be to check the K0S daughters
277  if (arrDaughLcIndex.size() == 3) {
278  for (auto iProng = 0; iProng < arrDaughLcIndex.size(); ++iProng) {
279  auto daughI = particlesMC.iteratorAt(arrDaughLcIndex[iProng]);
280  arrDaughLcPDG[iProng] = daughI.pdgCode();
281  }
282  if (!(arrDaughLcPDG[0] == arrDaughLcPDGRef[0] && arrDaughLcPDG[1] == arrDaughLcPDGRef[1] && arrDaughLcPDG[2] == arrDaughLcPDGRef[2])) { // this should be the condition, first bach, then v0
283  sign = 0;
284  } else {
285  LOG(INFO) << "Lc --> K0S+p found in MC table";
286  }
287  MY_DEBUG_MSG(sign == 0, LOG(INFO) << "Pity, the three final daughters are not p, pi+, pi-, but " << arrDaughLcPDG[0] << ", " << arrDaughLcPDG[1] << ", " << arrDaughLcPDG[2]);
288  }
289  }
290  rowMCMatchGen(sign);
291  }
292  }
293 };
