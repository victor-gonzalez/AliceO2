230 inline void AcceptTrack(aod::TrackData const& track, bool& asone, bool& astwo)
231 {
232 
233  asone = false;
234  astwo = false;
235 
236  /* TODO: incorporate a mask in the scanned tracks table for the rejecting track reason */
237  if (matchTrackType(track)) {
238  if (ptlow < track.pt() and track.pt() < ptup and etalow < track.eta() and track.eta() < etaup) {
239  if (((track.sign() > 0) and (trackonecharge > 0)) or ((track.sign() < 0) and (trackonecharge < 0))) {
240  asone = true;
241  }
242  if (((track.sign() > 0) and (tracktwocharge > 0)) or ((track.sign() < 0) and (tracktwocharge < 0))) {
243  astwo = true;
244  }
245  }
246  }

273 inline int GetEtaPhiIndex(aod::FilteredTrackData const& t)
274 {
275  int etaix = int((t.eta() - etalow) / etabinwidth);
276  /* consider a potential phi origin shift */
277  float phi = GetShiftedPhi(t.phi());
278  int phiix = int((phi - philow) / phibinwidth);
279  return etaix * phibins + phiix;

394  void process(aod::CollisionEvSelCent const& collision, soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra, aod::TracksExtended, aod::TrackSelection> const& ftracks)
395  {
396  using namespace filteranalyistask;
397 
398  // LOGF(INFO,"New collision with %d filtered tracks", ftracks.size());
399  fhCentMultB->Fill(collision.centV0M());
400  fhVertexZB->Fill(collision.posZ());
401  bool acceptedevent = false;
402  float centormult = -100.0;
403  if (IsEvtSelected(collision, centormult)) {
404  acceptedevent = true;
405  fhCentMultA->Fill(collision.centV0M());
406  fhVertexZA->Fill(collision.posZ());
407  // LOGF(INFO,"New accepted collision with %d filtered tracks", ftracks.size());
408 
409  for (auto& track : ftracks) {
410  /* before track selection */
411  fhPtB->Fill(track.pt());
412  fhEtaB->Fill(track.eta());
413  fhPhiB->Fill(track.phi());
414  fhEtaVsPhiB->Fill(track.phi(), track.eta());
415  fhPtVsEtaB->Fill(track.eta(), track.pt());
416  if (track.sign() > 0) {
417  fhPtPosB->Fill(track.pt());
418  } else {
419  fhPtNegB->Fill(track.pt());
420  }
421 
422  /* track selection */
423  /* tricky because the boolean columns issue */
424  bool asone, astwo;
425  AcceptTrack(track, asone, astwo);
426  if (asone or astwo) {
427  /* the track has been accepted */
428  fhPtA->Fill(track.pt());
429  fhEtaA->Fill(track.eta());
430  fhPhiA->Fill(track.phi());
431  fhEtaVsPhiA->Fill(track.phi(), track.eta());
432  fhPtVsEtaA->Fill(track.eta(), track.pt());
433  if (track.sign() > 0) {
434  fhPtPosA->Fill(track.pt());
435  } else {
436  fhPtNegA->Fill(track.pt());
437  }
438  }
439  scannedtracks((uint8_t)asone, (uint8_t)astwo);
440  }
441  } else {
442  for (auto& track : ftracks) {
443  scannedtracks((uint8_t) false, (uint8_t) false);
444  }
445  }
446  acceptedevents((uint8_t)acceptedevent, centormult);
447  }
