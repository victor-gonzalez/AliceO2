94  if (cutYCandMax >= 0. && std::abs(YDPlus(candidate1)) > cutYCandMax) {
95  continue;
96  }
97  if (cutPtCandMin >= 0. && candidate1.pt() < cutPtCandMin) {
98  continue;
99  }
100  //check decay channel flag for candidate1
101  if (!(candidate1.hfflag() & 1 << DecayType::DPlusToPiKPi)) { //probably dummy since already selected? not sure...
102  continue;
103  }
104  int outerParticleSign = 1; //Dplus
105  auto outerSecondTrack = candidate1.index1_as<aod::BigTracks>();
106  if (outerSecondTrack.sign() == 1) {
107  outerParticleSign = -1; //Dminus (second daughter track is positive)
108  }
109 
110  //fill invariant mass plots and generic info from all Dplus/Dminus candidates
111  if (outerParticleSign == 1) {
112  registry.fill(HIST("hMass"), InvMassDPlus(candidate1), candidate1.pt());
113  registry.fill(HIST("hMassDplus"), InvMassDPlus(candidate1), candidate1.pt());
114  } else {
115  registry.fill(HIST("hMass"), InvMassDPlus(candidate1), candidate1.pt());

117  }
118  registry.fill(HIST("hPtCand"), candidate1.pt());
119  registry.fill(HIST("hPtProng0"), candidate1.ptProng0());
120  registry.fill(HIST("hPtProng1"), candidate1.ptProng1());
121  registry.fill(HIST("hPtProng2"), candidate1.ptProng2());
122  registry.fill(HIST("hEta"), candidate1.eta());
123  registry.fill(HIST("hPhi"), candidate1.phi());
124  registry.fill(HIST("hY"), YDPlus(candidate1));
125  registry.fill(HIST("hSelectionStatus"), candidate1.isSelDplusToPiKPi());
126 
127  //D-Dbar correlation dedicated section
128  //if the candidate is a Dplus, search for Dminus and evaluate correlations
129  if (outerParticleSign != 1) {
130  continue;
131  }
132  for (auto& candidate2 : candidates) {
133  //check decay channel flag for candidate2
134  if (!(candidate2.hfflag() & 1 << DecayType::DPlusToPiKPi)) { //probably dummy since already selected? not sure...
135  continue;
136  }
137  auto innerSecondTrack = candidate2.index1_as<aod::BigTracks>();
138  if (innerSecondTrack.sign() != 1) { //keep only Dminus (with second daughter track positive)
139  continue;
140  }
141  if (cutYCandMax >= 0. && std::abs(YDPlus(candidate2)) > cutYCandMax) {
142  continue;
143  }
144  if (cutPtCandMin >= 0. && candidate2.pt() < cutPtCandMin) {
145  continue;
146  }
147  entryDplusDminusPair(getDeltaPhi(candidate2.phi(), candidate1.phi()),
148  candidate2.eta() - candidate1.eta(),
149  candidate1.pt(),
150  candidate2.pt());
151  entryDplusDminusRecoInfo(InvMassDPlus(candidate1),
152  InvMassDPlus(candidate2),
153  0);
154  double etaCut = 0.;
155  double ptCut = 0.;
156  do { //fill pairs vs etaCut plot
157  ptCut = 0.;

159  do { //fill pairs vs etaCut plot

161  registry.fill(HIST("hDDbarVsEtaCut"), etaCut - epsilon, ptCut + epsilon);
162  }
163  ptCut += incrementPtThreshold;
164  } while (ptCut < ptThresholdForMaxEtaCut - epsilon);
165  } while (etaCut < maxEtaCut - epsilon);
166  } // end inner loop (Dminus)
167  } //end outer loop (Dplus)
168  }

208  bool flagDminusSignal = false;

212  continue;
213  }

216  }

222  if (outerSecondTrack.sign() == 1) {
223  outerParticleSign = -1; //Dminus (second daughter track is positive)
224  }
225  if (std::abs(candidate1.flagMCMatchRec()) == 1 << DecayType::DPlusToPiKPi) {
226  //fill invariant mass plots and per-candidate distributions from Dplus/Dminus signal candidates
227  if (outerParticleSign == 1) { //reco and matched as Dplus
228  registry.fill(HIST("hMassDplusMCRecSig"), InvMassDPlus(candidate1), candidate1.pt());

237  registry.fill(HIST("hPhiMCRec"), candidate1.phi());
238  registry.fill(HIST("hYMCRec"), YDPlus(candidate1));
239  registry.fill(HIST("hSelectionStatusMCRec"), candidate1.isSelDplusToPiKPi());
240  } else {
241  //fill invariant mass plots from Dplus/Dminus background candidates
242  if (outerParticleSign == 1) { //reco as Dplus
243  registry.fill(HIST("hMassDplusMCRecBkg"), InvMassDPlus(candidate1), candidate1.pt());

251  continue; //reject Dminus in outer loop
252  }

254  for (auto& candidate2 : candidates) {
255  if (!(candidate2.hfflag() & 1 << DecayType::DPlusToPiKPi)) { //check decay channel flag for candidate2
256  continue;
257  }
258  auto innerSecondTrack = candidate2.index1_as<aod::BigTracks>();

261  }

295  } while (etaCut < maxEtaCut - epsilon);
296  } // end inner loop (Dbars)
297

327  void process(aod::McCollision const& mccollision, soa::Join<aod::McParticles, aod::HfCandProng3MCGen> const& particlesMC)
328  {
329  int counterDplusDminus = 0;
330  registry.fill(HIST("hMCEvtCount"), 0);
331  //MC gen level
332  for (auto& particle1 : particlesMC) {
333  //check if the particle is Dplus or Dminus (for general plot filling and selection, so both cases are fine) - NOTE: decay channel is not probed!
334  if (std::abs(particle1.pdgCode()) != pdg::Code::kDPlus) {
335  continue;
336  }
337  double yD = RecoDecay::Y(array{particle1.px(), particle1.py(), particle1.pz()}, RecoDecay::getMassPDG(particle1.pdgCode()));
338  if (cutYCandMax >= 0. && std::abs(yD) > cutYCandMax) {
339  continue;
340  }
341  if (cutPtCandMin >= 0. && particle1.pt() < cutPtCandMin) {
342  continue;
343  }
344  registry.fill(HIST("hPtCandMCGen"), particle1.pt());
345  registry.fill(HIST("hEtaMCGen"), particle1.eta());
346  registry.fill(HIST("hPhiMCGen"), particle1.phi());
347  registry.fill(HIST("hYMCGen"), yD);
348  counterDplusDminus++;
349 
350  //D-Dbar correlation dedicated section
351  //if it's a Dplus particle, search for Dminus and evaluate correlations
352  if (particle1.pdgCode() != pdg::Code::kDPlus) { //just checking the particle PDG, not the decay channel (differently from Reco: you have a BR factor btw such levels!)
353  continue;
354  }
355  registry.fill(HIST("hcountDplustriggersMCGen"), 0, particle1.pt()); //to count trigger Dplus (for normalisation)
356  for (auto& particle2 : particlesMC) {
357  if (particle2.pdgCode() != -pdg::Code::kDPlus) { //check that inner particle is a Dminus
358  continue;
359  }
360  if (cutYCandMax >= 0. && std::abs(RecoDecay::Y(array{particle2.px(), particle2.py(), particle2.pz()}, RecoDecay::getMassPDG(particle2.pdgCode()))) > cutYCandMax) {
361  continue;
362  }
363  if (cutPtCandMin >= 0. && particle2.pt() < cutPtCandMin) {
364  continue;
365  }
366  entryDplusDminusPair(getDeltaPhi(particle2.phi(), particle1.phi()),
367  particle2.eta() - particle1.eta(),
368  particle1.pt(),
369  particle2.pt());
370  double etaCut = 0.;
371  double ptCut = 0.;
372 
373  //fill pairs vs etaCut plot
374  bool rightDecayChannels = false;
375  if ((std::abs(particle1.flagMCMatchGen()) == 1 << DecayType::DPlusToPiKPi) && (std::abs(particle2.flagMCMatchGen()) == 1 << DecayType::DPlusToPiKPi)) {
376  rightDecayChannels = true;
377  }
378  do {
379  ptCut = 0.;
380  etaCut += incrementEtaCut;
381  do { //fill pairs vs etaCut plot
382  if (std::abs(particle1.eta()) < etaCut && std::abs(particle2.eta()) < etaCut && particle1.pt() > ptCut && particle2.pt() > ptCut) {
383  registry.fill(HIST("hDDbarVsEtaCut"), etaCut - epsilon, ptCut + epsilon);
384  }
385  if (rightDecayChannels) { //fill with D and Dbar daughter particls acceptance checks
386  double etaCandidate1Daughter1 = particlesMC.iteratorAt(particle1.daughter0()).eta();
387  double etaCandidate1Daughter2 = particlesMC.iteratorAt(particle1.daughter0() + 1).eta();

810  int counterccbar = 0, counterccbarPreEtasel = 0;

812  //loop over particles at MC gen level
813  for (auto& particle1 : particlesMC) {
814  if (std::abs(particle1.pdgCode()) != PDG_t::kCharm) { //search c or cbar particles
815  continue;
816  }
817  int partMothPDG = particlesMC.iteratorAt(particle1.mother0()).pdgCode();
818  //check whether mothers of quark c/cbar are still '4'/'-4' particles - in that case the c/cbar quark comes from its own fragmentation, skip it
819  if (partMothPDG == particle1.pdgCode()) {
820  continue;
821  }
822  counterccbarPreEtasel++; //count c or cbar (before kinematic selection)
823  double yC = RecoDecay::Y(array{particle1.px(), particle1.py(), particle1.pz()}, RecoDecay::getMassPDG(particle1.pdgCode()));
824  if (cutYCandMax >= 0. && std::abs(yC) > cutYCandMax) {
825  continue;
826  }
827  if (cutPtCandMin >= 0. && particle1.pt() < cutPtCandMin) {
828  continue;
829  }
830  registry.fill(HIST("hPtCandMCGen"), particle1.pt());
831  registry.fill(HIST("hEtaMCGen"), particle1.eta());
832  registry.fill(HIST("hPhiMCGen"), particle1.phi());
833  registry.fill(HIST("hYMCGen"), yC);
834  counterccbar++; //count if c or cbar don't come from themselves during fragmentation (after kinematic selection)
835 
836  //c-cbar correlation dedicated section
837  registry.fill(HIST("hcountCtriggersMCGen"), 0, particle1.pt()); //to count trigger c quark (for normalisation)
838 
839  for (auto& particle2 : particlesMC) {
840  if (std::abs(particle2.pdgCode()) != PDG_t::kCharm) { //search c or cbar for associated particles
841  continue;
842  }
843  if (cutYCandMax >= 0. && std::abs(RecoDecay::Y(array{particle2.px(), particle2.py(), particle2.pz()}, RecoDecay::getMassPDG(particle2.pdgCode()))) > cutYCandMax) {
844  continue;
845  }
846  if (cutPtCandMin >= 0. && particle2.pt() < cutPtCandMin) {
847  continue;
848  }
849  if (particle2.pdgCode() == particle1.pdgCode()) {
850  //check whether mothers of quark cbar (from associated loop) are still '-4' particles - in that case the cbar quark comes from its own fragmentation, skip it
851  if (particlesMC.iteratorAt(particle2.mother0()).pdgCode() == particle2.pdgCode()) {
852  continue;
853  }
854  //Excluding self-correlations (in principle not possible due to the '<' condition, but could rounding break it?)
855  if (particle1.mRowIndex == particle2.mRowIndex) {
856  continue;
857  }
858  entryccbarPair(getDeltaPhi(particle2.phi(), particle1.phi()),
859  particle2.eta() - particle1.eta(),
860  particle1.pt(),
861  particle2.pt());
862  } // end outer if (check PDG associate)
863  } // end inner loop
864  } //end outer loop
865  registry.fill(HIST("hcountCCbarPerEvent"), counterccbar);
866  registry.fill(HIST("hcountCCbarPerEventPreEtaCut"), counterccbarPreEtasel);
