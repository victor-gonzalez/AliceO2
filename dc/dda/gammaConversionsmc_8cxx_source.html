160  template <typename TTRACKS, typename TCOLL, typename TV0>
161  bool processPhoton(TCOLL const& theCollision, const TV0& theV0)
162  {
163  fillHistogramsBeforeCuts(theV0);
164 
165  auto lTrackPos = theV0.template posTrack_as<TTRACKS>(); //positive daughter
166  auto lTrackNeg = theV0.template negTrack_as<TTRACKS>(); //negative daughter
167 
168  // apply track cuts
169  if (!(trackPassesCuts(lTrackPos) && trackPassesCuts(lTrackNeg))) {
170  return kFALSE;
171  }
172 
173  float lV0CosinePA = theV0.v0cosPA(theCollision.posX(), theCollision.posY(), theCollision.posZ());
174 
175  // apply photon cuts
176  if (!passesPhotonCuts(theV0, lV0CosinePA)) {
177  return kFALSE;
178  }
179 
180  fillHistogramsAfterCuts(theV0, lTrackPos, lTrackNeg, lV0CosinePA);
181 
182  return kTRUE;
183  }

185  template <typename TV0, typename TMC>
186  void processTruePhoton(const TV0& theV0, const TMC& theMcParticles)
187  {
188  auto lTrackPos = theV0.template posTrack_as<tracksAndTPCInfoMC>(); //positive daughter
189  auto lTrackNeg = theV0.template negTrack_as<tracksAndTPCInfoMC>(); //negative daughter
190 
191  // todo: verify it is enough to check only mother0 being equal
192  if (lTrackPos.mcParticle().mother0() > -1 &&
193  lTrackPos.mcParticle().mother0() == lTrackNeg.mcParticle().mother0()) {
194  auto lMother = theMcParticles.iteratorAt(lTrackPos.mcParticle().mother0());
195 
196  if (lMother.pdgCode() == 22) {
197 
198  registry.fill(HIST("resolutions/hPtRes"), theV0.pt() - lMother.pt());
199  registry.fill(HIST("resolutions/hEtaRes"), theV0.eta() - lMother.eta());
200  registry.fill(HIST("resolutions/hPhiRes"), theV0.phi() - lMother.phi());
201 
202  TVector3 lConvPointRec(theV0.x(), theV0.y(), theV0.z());
203  TVector3 lPosTrackVtxMC(lTrackPos.mcParticle().vx(), lTrackPos.mcParticle().vy(), lTrackPos.mcParticle().vz());
204  // take the origin of the positive mc track as conversion point (should be identical with negative, verified this on a few photons)
205  TVector3 lConvPointMC(lPosTrackVtxMC);
206 
207  registry.fill(HIST("resolutions/hConvPointRRes"), lConvPointRec.Perp() - lConvPointMC.Perp());
208  registry.fill(HIST("resolutions/hConvPointAbsoluteDistanceRes"), TVector3(lConvPointRec - lConvPointMC).Mag());
209  }
210  }

213  void processMC(collisionEvSelIt const& theCollision,
214  aod::V0Datas const& theV0s,
215  tracksAndTPCInfoMC const& theTracks,
216  aod::McParticles const& theMcParticles)
217  {
218  registry.fill(HIST("hEvents"), 0);
219  if (fDoEventSel && !theCollision.sel7()) {
220  registry.fill(HIST("hEvents"), 1);
221  return;
222  }
223  registry.fill(HIST("hEvents"), 3);
224 
225  for (auto& lV0 : theV0s) {
226  if (!processPhoton<tracksAndTPCInfoMC>(theCollision, lV0)) {
227  continue;
228  }
229  processTruePhoton(lV0, theMcParticles);
230  }

233  template <typename T>
234  void fillHistogramsBeforeCuts(const T& theV0)
235  {
236  // fill some QA histograms before any cuts
237  registry.fill(HIST("beforeCuts/hPtRec"), theV0.pt());
238  registry.fill(HIST("beforeCuts/hEtaRec"), theV0.eta());
239  registry.fill(HIST("beforeCuts/hPhiRec"), theV0.phi());

244  template <typename T>
245  bool trackPassesCuts(const T& theTrack)
246  {
247 
248  // single track eta cut
249  if (TMath::Abs(theTrack.eta()) > fEtaCut) {
250  registry.fill(HIST("IsPhotonSelected"), kTrackEta);
251  return kFALSE;
252  }
253 
254  // single track pt cut
255  if (theTrack.pt() < fSinglePtCut) {
256  registry.fill(HIST("IsPhotonSelected"), kTrackPt);
257  return kFALSE;
258  }
259 
260  if (!(selectionPIDTPC_track(theTrack))) {
261  return kFALSE;
262  }
263 
264  if (theTrack.tpcFoundOverFindableCls() < fMinTPCFoundOverFindableCls) {
265  registry.fill(HIST("IsPhotonSelected"), kTPCFoundOverFindableCls);
266  return kFALSE;
267  }
268 
269  if (theTrack.tpcCrossedRowsOverFindableCls() < fMinTPCCrossedRowsOverFindableCls) {
270  registry.fill(HIST("IsPhotonSelected"), kTPCCrossedRowsOverFindableCls);
271  return kFALSE;
272  }
273 
274  return kTRUE;
275  }
276 
277  template <typename T>
278  bool passesPhotonCuts(const T& theV0, float theV0CosinePA)
279  {
280  if (theV0.v0radius() < fMinR || theV0.v0radius() > fMaxR) {
281  registry.fill(HIST("IsPhotonSelected"), kV0Radius);
282  return kFALSE;
283  }
284 
285  if (!ArmenterosQtCut(theV0.alpha(), theV0.qtarm(), theV0.pt())) {
286  registry.fill(HIST("IsPhotonSelected"), kArmenteros);
287  return kFALSE;
288  }
289 
290  if (TMath::Abs(theV0.psipair()) > fPsiPairCut) {
291  registry.fill(HIST("IsPhotonSelected"), kPsiPair);
292  return kFALSE;
293  }
294 
295  if (theV0CosinePA < fCosPAngleCut) {
296  registry.fill(HIST("IsPhotonSelected"), kCosinePA);
297  return kFALSE;
298  }
299 
300  return kTRUE;
301  }

303  template <typename TV0, typename TTRACK>
304  void fillHistogramsAfterCuts(const TV0& theV0, const TTRACK& theTrackPos, const TTRACK& theTrackNeg, float theV0CosinePA)
305  {

308  registry.fill(HIST("hPtRec"), theV0.pt());
309  registry.fill(HIST("hEtaRec"), theV0.eta());
310  registry.fill(HIST("hPhiRec"), theV0.phi());
311  registry.fill(HIST("hConvPointR"), theV0.v0radius());
312 
313  registry.fill(HIST("hTPCdEdxSigEl"), theTrackNeg.p(), theTrackNeg.tpcNSigmaEl());
314  registry.fill(HIST("hTPCdEdxSigEl"), theTrackPos.p(), theTrackPos.tpcNSigmaEl());
315  registry.fill(HIST("hTPCdEdxSigPi"), theTrackNeg.p(), theTrackNeg.tpcNSigmaPi());
316  registry.fill(HIST("hTPCdEdxSigPi"), theTrackPos.p(), theTrackPos.tpcNSigmaPi());
317 
318  registry.fill(HIST("hTPCdEdx"), theTrackNeg.p(), theTrackNeg.tpcSignal());
319  registry.fill(HIST("hTPCdEdx"), theTrackPos.p(), theTrackPos.tpcSignal());
320 
321  registry.fill(HIST("hTPCFoundOverFindableCls"), theTrackNeg.tpcFoundOverFindableCls());
322  registry.fill(HIST("hTPCFoundOverFindableCls"), theTrackPos.tpcFoundOverFindableCls());
323 
324  registry.fill(HIST("hTPCCrossedRowsOverFindableCls"), theTrackNeg.tpcCrossedRowsOverFindableCls());
325  registry.fill(HIST("hTPCCrossedRowsOverFindableCls"), theTrackPos.tpcCrossedRowsOverFindableCls());
326 
327  registry.fill(HIST("hArmenteros"), theV0.alpha(), theV0.qtarm());
328  registry.fill(HIST("hPsiPtRec"), theV0.psipair(), theV0.pt());
329 
330  registry.fill(HIST("hCosPAngle"), theV0CosinePA);
331  }

333  Bool_t ArmenterosQtCut(Double_t theAlpha, Double_t theQt, Double_t thePt)
334  {
335  // in AliPhysics this is the cut for if fDo2DQt && fDoQtGammaSelection == 2
336  Float_t lQtMaxPtDep = fQtPtMax * thePt;
337  if (lQtMaxPtDep > fQtMax) {
338  lQtMaxPtDep = fQtMax;
339  }
340  if (!(TMath::Power(theAlpha / fMaxPhotonAsymmetry, 2) + TMath::Power(theQt / lQtMaxPtDep, 2) < 1)) {
341  return kFALSE;
342  }
343  return kTRUE;
344  }

346  template <typename T>
347  bool selectionPIDTPC_track(const T& theTrack)
348  {
349  // TPC Electron Line
350  if (theTrack.tpcNSigmaEl() < fPIDnSigmaBelowElectronLine || theTrack.tpcNSigmaEl() > fPIDnSigmaAboveElectronLine) {
351  registry.fill(HIST("IsPhotonSelected"), kElectronPID);
352  return kFALSE;
353  }
354 
355  // TPC Pion Line
356  if (theTrack.p() > fPIDMinPnSigmaAbovePionLine) {
357  // low pt Pion rej
358  if (theTrack.p() < fPIDMaxPnSigmaAbovePionLine) {
359  if (theTrack.tpcNSigmaEl() > fPIDnSigmaBelowElectronLine && theTrack.tpcNSigmaEl() < fPIDnSigmaAboveElectronLine && theTrack.tpcNSigmaPi() < fPIDnSigmaAbovePionLine) {
360  registry.fill(HIST("IsPhotonSelected"), kPionRejLowMom);
361  return kFALSE;
362  }
363  }
364  // High Pt Pion rej
365  else {
366  if (theTrack.tpcNSigmaEl() > fPIDnSigmaBelowElectronLine && theTrack.tpcNSigmaEl() < fPIDnSigmaAboveElectronLine && theTrack.tpcNSigmaPi() < fPIDnSigmaAbovePionLineHighP) {
367  registry.fill(HIST("IsPhotonSelected"), kPionRejHighMom);
368  return kFALSE;
369  }
370  }
371  }
372  return kTRUE;
373  }
