67  void process(const soa::Join<o2::aod::Tracks, o2::aod::McTrackLabels>& tracks,
68  const aod::McParticles& mcParticles)
69  {
70 
71  std::vector<int64_t> ParticlesOfInterest;
72  for (const auto& mcParticle : mcParticles) {
73  if (mcParticle.pdgCode() != PDG) {
74  continue;
75  }
76  if (mcParticle.y() < yMin || mcParticle.y() > yMax) {
77  continue;
78  }
79  histos.fill(HIST("particlePt"), mcParticle.pt());
80  histos.fill(HIST("particleEta"), mcParticle.eta());
81  histos.fill(HIST("particleY"), mcParticle.y());
82  histos.fill(HIST("prodVx"), mcParticle.vx());
83  histos.fill(HIST("prodVy"), mcParticle.vy());
84  histos.fill(HIST("prodVz"), mcParticle.vz());
85  histos.fill(HIST("prodRadius"), std::sqrt(mcParticle.vx() * mcParticle.vx() + mcParticle.vy() * mcParticle.vy()));
86  histos.fill(HIST("prodVxVsPt"), mcParticle.pt(), mcParticle.vx());
87  histos.fill(HIST("prodVyVsPt"), mcParticle.pt(), mcParticle.vy());
88  histos.fill(HIST("prodVzVsPt"), mcParticle.pt(), mcParticle.vz());
89  histos.fill(HIST("prodRadiusVsPt"), mcParticle.pt(), std::sqrt(mcParticle.vx() * mcParticle.vx() + mcParticle.vy() * mcParticle.vy()));
90  histos.fill(HIST("prodRadius3DVsPt"), mcParticle.pt(), std::sqrt(mcParticle.vx() * mcParticle.vx() + mcParticle.vy() * mcParticle.vy() + mcParticle.vz() * mcParticle.vz()));
91  ParticlesOfInterest.push_back(mcParticle.globalIndex());
92  }
93 
94  for (const auto& track : tracks) {
95  const auto mcParticle = track.mcParticle();
96  if (!IsStable) {
97  if (mcParticle.mother0() < 0) {
98  continue;
99  }
100  auto mother = mcParticles.iteratorAt(mcParticle.mother0());
101  const auto ParticleIsInteresting = std::find(ParticlesOfInterest.begin(), ParticlesOfInterest.end(), mother.globalIndex()) != ParticlesOfInterest.end();
102  if (!ParticleIsInteresting) {
103  continue;
104  }
105  if (doPrint) {
106  LOG(INFO) << "Track " << track.globalIndex() << " comes from a " << mother.pdgCode() << " and is a " << mcParticle.pdgCode();
107  }
108  } else {
109  if (mcParticle.pdgCode() != PDG) {
110  continue;
111  }
112  histos.fill(HIST("trackPt"), track.pt() * charge);
113  histos.fill(HIST("trackEta"), track.eta());
114  if (mcParticle.mother0() < 0) {
115  if (doPrint) {
116  LOG(INFO) << "Track " << track.globalIndex() << " is a " << mcParticle.pdgCode();
117  }
118  continue;
119  }
120  auto mother = mcParticles.iteratorAt(mcParticle.mother0());
121  if (MC::isPhysicalPrimary(mcParticles, mcParticle)) {
122  histos.get<TH1>(HIST("primaries"))->Fill(Form("%i", mother.pdgCode()), 1.f);
123  } else {
124  histos.get<TH1>(HIST("secondaries"))->Fill(Form("%i", mother.pdgCode()), 1.f);
125  }
126  if (doPrint) {
127  LOG(INFO) << "Track " << track.globalIndex() << " is a " << mcParticle.pdgCode() << " and comes from a " << mother.pdgCode() << " and is " << (MC::isPhysicalPrimary(mcParticles, mcParticle) ? "" : "not") << " a primary";
128  }
129  }
130  }
131  }
132 };
