58  void run(o2::framework::ProcessingContext& pc) final
59  {
60  // set the min range of TFs for first TF
61  if (mProcessedTFs == 0) {
62  mTFRange[0] = getCurrentTF(pc);
63  }
64 
65  // write struct containing grouping parameters to access grouped IDCs to CCDB
66  if (mWriteToDB && mUpdateGroupingPar) {
67  // validity for grouping parameters is from first TF to some really large TF (until it is updated)
68  mDBapi.storeAsTFileAny<o2::tpc::ParameterIDCGroupCCDB>(&mIDCFactorization.getGroupingParameter(), "TPC/Calib/IDC/GROUPINGPAR", mMetadata, getFirstTF(), std::numeric_limits<uint32_t>::max());
69  mUpdateGroupingPar = false; // write grouping parameters only once
70  }
71 
72  for (int i = 0; i < 2 * mCRUs.size(); ++i) {
73  const DataRef ref = pc.inputs().getByPos(i);
74  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
75  const int cru = tpcCRUHeader->subSpecification >> 7;
76  const auto descr = tpcCRUHeader->dataDescription;
77  if (TPCAverageGroupIDCDevice::getDataDescription1DIDC() == descr) {
78  // 1D-IDCs as input which will be used for FT
79  const o2::tpc::CRU cruTmp(cru);
80  mOneDIDCAggregator.aggregate1DIDCs(cruTmp.side(), pc.inputs().get<std::vector<float>>(ref), mProcessedTFs, cruTmp.region());
81  } else if (TPCAverageGroupIDCDevice::getDataDescriptionIDCGroup() == descr) {
82  // 3D-IDCs as input which will be factorized
83  mIDCFactorization.setIDCs(pc.inputs().get<std::vector<float>>(ref), cru, mProcessedTFs); // aggregate IDCs
84  } else {
85  // wrong description;
86  }
87  }
88  ++mProcessedTFs;
89 
90  if (mProcessedTFs == mIDCFactorization.getNTimeframes()) {
91  mTFRange[1] = getCurrentTF(pc); // set the TF for last aggregated TF
92  mProcessedTFs = 0; // reset processed TFs for next aggregation interval
93  mIDCFactorization.factorizeIDCs(); // calculate DeltaIDC, 0D-IDC, 1D-IDC
94 
95  // perform fourier transform of 1D-IDCs
96  mIDCFourierTransform.setIDCs(std::move(mOneDIDCAggregator).getAggregated1DIDCs(), mIDCFactorization.getIntegrationIntervalsPerTF());
97  mIDCFourierTransform.calcFourierCoefficients();
98 
99  if (mDebug) {
100  LOGP(info, "dumping aggregated and factorized IDCs and FT to file");
101  mIDCFactorization.dumpToFile(fmt::format("IDCFactorized_{:02}.root", getCurrentTF(pc)).data());
102  mIDCFourierTransform.dumpToFile(fmt::format("Fourier_{:02}.root", getCurrentTF(pc)).data());
103  }
104 
105  // storing to CCDB
106  sendOutput(pc.outputs());
107  }
108  }

130  uint32_t getCurrentTF(o2::framework::ProcessingContext& pc) const { return o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getByPos(0))->tfCounter; }
131 
133  uint32_t getFirstTF() const { return mTFRange[0]; }
134 
136  uint32_t getLastTF() const { return mTFRange[1]; }
137 
139  unsigned int getFirstTFDeltaIDC(const unsigned int iChunk) const { return getFirstTF() + iChunk * mIDCFactorization.getTimeFramesDeltaIDC(); }
140 
142  unsigned int getLastTFDeltaIDC(const unsigned int iChunk) const { return (iChunk == mIDCFactorization.getNChunks() - 1) ? (mIDCFactorization.getNTimeframes() - 1 + getFirstTF()) : (getFirstTFDeltaIDC(iChunk) + mIDCFactorization.getTimeFramesDeltaIDC() - 1); }
143 
144  void sendOutput(DataAllocator& output)
145  {
146  if (mWriteToDB) {
147  const long timeStampStart = getFirstTF();
148  const long timeStampEnd = getLastTF();
149  mDBapi.storeAsTFileAny<o2::tpc::IDCZero>(&mIDCFactorization.getIDCZero(), "TPC/Calib/IDC/IDC0", mMetadata, timeStampStart, timeStampEnd);
150  mDBapi.storeAsTFileAny<o2::tpc::IDCOne>(&mIDCFactorization.getIDCOne(), "TPC/Calib/IDC/IDC1", mMetadata, timeStampStart, timeStampEnd);
151  mDBapi.storeAsTFileAny<o2::tpc::FourierCoeff>(&mIDCFourierTransform.getFourierCoefficients(), "TPC/Calib/IDC/FOURIER", mMetadata, timeStampStart, timeStampEnd);
152 
153  for (unsigned int iChunk = 0; iChunk < mIDCFactorization.getNChunks(); ++iChunk) {
154  switch (mCompressionDeltaIDC) {
155  case IDCDeltaCompression::MEDIUM:
156  default: {
157  auto idcDeltaMediumCompressed = mIDCFactorization.getIDCDeltaMediumCompressed(iChunk);
158  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<short>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
159  break;
160  }
161  case IDCDeltaCompression::HIGH: {
162  auto idcDeltaHighCompressed = mIDCFactorization.getIDCDeltaHighCompressed(iChunk);
163  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<char>>(&idcDeltaHighCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
164  break;
165  }
166  case IDCDeltaCompression::NO:
167  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<float>>(&mIDCFactorization.getIDCDeltaUncompressed(iChunk), "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
168  break;
169  }
170  }
171  }
172  // reseting aggregated IDCs. This is done for safety, but if all data is received in the next aggregation interval it isnt necessary... remove it?
173  mIDCFactorization.reset();
174  }

177 DataProcessorSpec getTPCAggregateGroupedIDCSpec(const std::vector<uint32_t>& crus, const unsigned int timeframes, const unsigned int timeframesDeltaIDC, const unsigned int rangeIDC, const unsigned int nFourierCoefficientsStore, const IDCDeltaCompression compression, const bool debug = false)
178 {
179  std::vector<OutputSpec> outputSpecs;
180  std::vector<InputSpec> inputSpecs;
181  inputSpecs.reserve(crus.size());
182  for (const auto& cru : crus) {
183  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
184  inputSpecs.emplace_back(InputSpec{"idcsgroup", gDataOriginTPC, TPCAverageGroupIDCDevice::getDataDescriptionIDCGroup(), subSpec, Lifetime::Timeframe});
185  inputSpecs.emplace_back(InputSpec{"1didc", gDataOriginTPC, TPCAverageGroupIDCDevice::getDataDescription1DIDC(), subSpec, Lifetime::Timeframe});
186  }
187 
188  const auto& paramIDCGroup = ParameterIDCGroup::Instance();
189  std::array<unsigned char, Mapper::NREGIONS> groupPads{};
190  std::array<unsigned char, Mapper::NREGIONS> groupRows{};
191  std::array<unsigned char, Mapper::NREGIONS> groupLastRowsThreshold{};
192  std::array<unsigned char, Mapper::NREGIONS> groupLastPadsThreshold{};
193  std::copy(std::begin(paramIDCGroup.GroupPads), std::end(paramIDCGroup.GroupPads), std::begin(groupPads));
194  std::copy(std::begin(paramIDCGroup.GroupRows), std::end(paramIDCGroup.GroupRows), std::begin(groupRows));
195  std::copy(std::begin(paramIDCGroup.GroupLastRowsThreshold), std::end(paramIDCGroup.GroupLastRowsThreshold), std::begin(groupLastRowsThreshold));
196  std::copy(std::begin(paramIDCGroup.GroupLastPadsThreshold), std::end(paramIDCGroup.GroupLastPadsThreshold), std::begin(groupLastPadsThreshold));
197 
198  return DataProcessorSpec{
199  "tpc-aggregate-idc",
200  inputSpecs,
201  outputSpecs,
202  AlgorithmSpec{adaptFromTask<TPCAggregateGroupedIDCSpec>(crus, timeframes, nFourierCoefficientsStore, timeframesDeltaIDC, groupPads, groupRows, groupLastRowsThreshold, groupLastPadsThreshold, rangeIDC, compression, debug)},
203  Options{{"ccdb-uri", VariantType::String, "http://ccdb-test.cern.ch:8080", {"URI for the CCDB access."}},
204  {"update-not-grouping-parameter", VariantType::Bool, false, {"Do NOT Update/Writing grouping parameters to CCDB."}}}}; // end DataProcessorSpec
205 }
