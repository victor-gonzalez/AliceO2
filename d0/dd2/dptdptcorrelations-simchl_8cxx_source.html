171  /* TODO: vertex quality checks */

177  return true;

181  break;
182  }
183  return false;
184  }

209 {
210 
211  asone = false;
212  astwo = false;
213 
214  /* TODO: incorporate a mask in the scanned tracks table for the rejecting track reason */
215  if (matchTrackType(track)) {
216  if (ptlow < track.pt() and track.pt() < ptup and etalow < track.eta() and track.eta() < etaup) {
217  if (((track.sign() > 0) and (trackonecharge > 0)) or ((track.sign() < 0) and (trackonecharge < 0))) {
218  asone = true;
219  }
220  if (((track.sign() > 0) and (tracktwocharge > 0)) or ((track.sign() < 0) and (tracktwocharge < 0))) {
221  astwo = true;

336  void process(aod::CollisionEvSelCent const& collision, soa::Join<aod::Tracks, aod::TracksCov, aod::TracksExtra, aod::TracksExtended, aod::TrackSelection> const& ftracks)
337  {
338  using namespace filteranalyistask;
339 
340  // LOGF(INFO,"New collision with %d filtered tracks", ftracks.size());
341  fhCentMultB->Fill(collision.centV0M());
342  fhVertexZB->Fill(collision.posZ());
343  bool acceptedevent = false;
344  float centormult = -100.0;
345  if (IsEvtSelected(collision, centormult)) {
346  acceptedevent = true;
347  fhCentMultA->Fill(collision.centV0M());
348  fhVertexZA->Fill(collision.posZ());
349  // LOGF(INFO,"New accepted collision with %d filtered tracks", ftracks.size());
350 
351  for (auto& track : ftracks) {
352  /* before track selection */
353  fhPtB->Fill(track.pt());
354  fhEtaB->Fill(track.eta());
355  fhPhiB->Fill(track.phi());
356  fhEtaVsPhiB->Fill(track.phi(), track.eta());
357  fhPtVsEtaB->Fill(track.eta(), track.pt());
358  if (track.sign() > 0) {
359  fhPtPosB->Fill(track.pt());
360  } else {
361  fhPtNegB->Fill(track.pt());
362  }
363 
364  /* track selection */
365  /* tricky because the boolean columns issue */
366  bool asone, astwo;
367  AcceptTrack(track, asone, astwo);
368  if (asone or astwo) {
369  /* the track has been accepted */
370  fhPtA->Fill(track.pt());
371  fhEtaA->Fill(track.eta());
372  fhPhiA->Fill(track.phi());
373  fhEtaVsPhiA->Fill(track.phi(), track.eta());
374  fhPtVsEtaA->Fill(track.eta(), track.pt());
375  if (track.sign() > 0) {
376  fhPtPosA->Fill(track.pt());
377  } else {
378  fhPtNegA->Fill(track.pt());
379  }
380  }
381  scannedtracks((uint8_t)asone, (uint8_t)astwo);
382  }
383  } else {
384  for (auto& track : ftracks) {
385  scannedtracks((uint8_t) false, (uint8_t) false);
386  }
387  }
388  acceptedevents((uint8_t)acceptedevent, centormult);
389  }

468  template <typename TrackObject>

540  template <typename TrackListObject>
541  void processTrackPairs(TrackListObject const& trks1, TrackListObject const& trks2, int pix, float cmul)
542  {
543  /* process pair magnitudes */
544  double n2 = 0;
545  double sum2PtPt = 0;
546  double sum2DptDpt = 0;
547  double n2nw = 0;
548  double sum2PtPtnw = 0;
549  double sum2DptDptnw = 0;
550  for (auto& track1 : trks1) {
551  double ptavg_1 = 0.0; /* TODO: load ptavg_1 for eta1, phi1 bin */
552  double corr1 = 1.0; /* TODO: track correction weights */
553  for (auto& track2 : trks2) {
554  /* checkiing the same track id condition */
555  if (track1 == track2) {
556  /* exclude autocorrelations */
557  continue;
558  } else {
559  /* process pair magnitudes */
560  double ptavg_2 = 0.0; /* TODO: load ptavg_2 for eta2, phi2 bin */
561  double corr2 = 1.0; /* TODO: track correction weights */
562  double corr = corr1 * corr2;
563  double dptdpt = (track1.pt() - ptavg_1) * (track2.pt() - ptavg_2);
564  n2 += corr;
565  sum2PtPt += track1.pt() * track2.pt() * corr;
566  sum2DptDpt += corr * dptdpt;
567  n2nw += 1;
568  sum2PtPtnw += track1.pt() * track2.pt();
569  sum2DptDptnw += dptdpt;
570  /* get the global bin for filling the differential histograms */
