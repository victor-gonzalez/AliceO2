64 bool DataReaderTask::isTimeFrameEmpty(ProcessingContext& pc)
65 {
66  constexpr auto origin = header::gDataOriginTRD;
67  o2::framework::InputSpec dummy{"dummy",
68  framework::ConcreteDataMatcher{origin,
69  header::gDataDescriptionRawData,
70  0xDEADBEEF}};
71  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload.
72  // frame detected we have no data and send this instead
73  // send empty output so as to not block workflow
74  for (const auto& ref : o2::framework::InputRecordWalker(pc.inputs(), {dummy})) {
75  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
76  if (dh->payloadSize == 0) {
77  LOGP(INFO, "Found blank input input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : ",
78  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize);
79  return true;
80  }
81  }
82  return false;
83 }

85 void DataReaderTask::run(ProcessingContext& pc)
86 {
87  LOG(info) << "TRD Translator Task run";
88  auto dataReadStart = std::chrono::high_resolution_clock::now();
89 
90  if (isTimeFrameEmpty(pc)) {
91  sendData(pc, true); //send the empty tf data.
92  return;
93  }
94  /* set encoder output buffer */
95  char bufferOut[o2::trd::constants::HBFBUFFERMAX];
96  int loopcounter = 0;
97  auto device = pc.services().get<o2::framework::RawDeviceService>().device();
98  auto outputRoutes = pc.services().get<o2::framework::RawDeviceService>().spec().outputs;
99  auto fairMQChannel = outputRoutes.at(0).channel;
100  /* loop over inputs routes */
101  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
102  if (!iit.isValid()) {
103  continue;
104  }
105  /* loop over input parts */
106  int inputpartscount = 0;
107  int emptyframe = 0;
108  for (auto const& ref : iit) {
109  if (mVerbose) {
110  const auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
111  LOGP(info, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : ",
112  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize);
113  }
114  const auto* headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
115  auto payloadIn = ref.payload;
116  auto payloadInSize = headerIn->payloadSize;
117  const auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
118  if (std::string(headerIn->dataDescription.str) != std::string("DISTSUBTIMEFRAMEFLP")) {
119  if (!mCompressedData) { //we have raw data coming in from flp
120  if (mVerbose) {
121  LOG(info) << " parsing non compressed data in the data reader task with a payload of " << payloadInSize << " payload size";
122  }
123  mReader.setDataBuffer(payloadIn);
124  mReader.setDataBufferSize(payloadInSize);
125  mReader.configure(mByteSwap, mFixDigitEndCorruption, mVerbose, mHeaderVerbose, mDataVerbose);
126  mReader.run();
127  if (mVerbose) {
128  LOG(info) << "relevant vectors to read : " << mReader.sumTrackletsFound() << " tracklets and " << mReader.sumDigitsFound() << " compressed digits";
129  }
130  // mTriggers = mReader.getIR();
131  } else { // we have compressed data coming in from flp.
132  mCompressedReader.setDataBuffer(payloadIn);
133  mCompressedReader.setDataBufferSize(payloadInSize);
134  mCompressedReader.configure(mByteSwap, mVerbose, mHeaderVerbose, mDataVerbose);
135  mCompressedReader.run();
136  }
137  } // ignore the input of DISTSUBTIMEFRAMEFLP
138  }
139  /* output */
140  sendData(pc, false);
141  }
142 
143  auto dataReadTime = std::chrono::high_resolution_clock::now() - dataReadStart;
144  LOG(info) << "Processing time for Data reading " << std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count() << "ms";
145  if (!mCompressedData) {
146  LOG(info) << "Digits found : " << mReader.getDigitsFound();
147  LOG(info) << "Tracklets found : " << mReader.getTrackletsFound();
148  }
149 }
