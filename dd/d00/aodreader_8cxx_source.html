120 struct aodReader {
121  void process(aod::Eins const& unos, aod::Zwei const& dues, aod::Drei const& tres)
122  {
123  int cnt = 0;
124  for (auto& uno : unos) {
125  auto eta = uno.eta();
126  auto mom = uno.mom();
127 
128  cnt++;
129  LOGF(INFO, "Eins (%i): (%f, %f)", cnt, eta, mom);
130  }
131  LOGF(INFO, "ATask Processed %i data points from Eins", cnt);
132 
133  cnt = 0;
134  for (auto& due : dues) {
135  auto ok = due.ok();
136  auto phi = due.phi();
137  auto pt = due.pt();
138 
139  cnt++;
140  LOGF(INFO, "Zwei (%i): (%i, %f, %f)", cnt, ok, phi, pt);
141  }
142  LOGF(INFO, "ATask Processed %i data points from Zwei", cnt);
143 
144  cnt = 0;
145  for (auto& tre : tres) {
146  auto eta = tre.eta();
147  auto phi = tre.phi();
148  auto mom = tre.mom();
149 
150  cnt++;
151  LOGF(INFO, "Drei (%i): (%f, %f, %f)", cnt, eta, phi, mom);
152  }
153  LOGF(INFO, "ATask Processed %i data points from Drei", cnt);
154  }
155 };
