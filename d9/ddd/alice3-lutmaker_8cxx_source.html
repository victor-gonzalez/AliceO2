41 template <o2::track::pid_constants::ID particle>
42 struct Alice3LutMaker {
43  static constexpr int nSpecies = 8;
44  static constexpr int PDGs[nSpecies] = {kElectron, kMuonMinus, kPiPlus, kKPlus, kProton, 1000010020, 1000010030, 1000020030};
45  static_assert(particle < nSpecies && "Maximum of particles reached");
46  static constexpr int pdg = PDGs[particle];
47  Configurable<bool> selPrim{"sel-prim", false, "If true selects primaries, if not select all particles"};
48  Configurable<int> etaBins{"eta-bins", 80, "Number of eta bins"};
49  Configurable<float> etaMin{"eta-min", -4.f, "Lower limit in eta"};
50  Configurable<float> etaMax{"eta-max", 4.f, "Upper limit in eta"};
51  Configurable<int> ptBins{"pt-bins", 200, "Number of pT bins"};
52  Configurable<float> ptMin{"pt-min", -2.f, "Lower limit in pT"};
53  Configurable<float> ptMax{"pt-max", 2.f, "Upper limit in pT"};
54  Configurable<int> logPt{"log-pt", 0, "Flag to use a logarithmic pT axis, in this case the pT limits are the expontents"};
55  HistogramRegistry histos{"Histos", {}, OutputObjHandlingPolicy::AnalysisObject};
56 
57  void init(InitContext&)
58  {
59  const TString commonTitle = Form(" PDG %i", pdg);
60  AxisSpec axisPt{ptBins, ptMin, ptMax, "#it{p}_{T} GeV/#it{c}"};
61  if (logPt) {
62  const double min = axisPt.binEdges[0];
63  const double width = (axisPt.binEdges[1] - axisPt.binEdges[0]) / axisPt.nBins.value();
64  axisPt.binEdges.clear();
65  axisPt.binEdges.resize(0);
66  for (int i = 0; i < axisPt.nBins.value() + 1; i++) {
67  axisPt.binEdges.push_back(std::pow(10., min + i * width));
68  }
69  axisPt.nBins = std::nullopt;
70  }
71  const AxisSpec axisEta{etaBins, etaMin, etaMax, "#it{#eta}"};
72 
73  histos.add("multiplicity", "Track multiplicity;Tracks per event;Events", kTH1F, {{100, 0, 2000}});
74  histos.add("pt", "pt" + commonTitle, kTH1F, {axisPt});
75  histos.add("eta", "eta" + commonTitle, kTH1F, {axisEta});
76  histos.add("CovMat_cYY", "cYY" + commonTitle, kTProfile2D, {axisPt, axisEta});
77  histos.add("CovMat_cZY", "cZY" + commonTitle, kTProfile2D, {axisPt, axisEta});
78  histos.add("CovMat_cZZ", "cZZ" + commonTitle, kTProfile2D, {axisPt, axisEta});
79  histos.add("CovMat_cSnpY", "cSnpY" + commonTitle, kTProfile2D, {axisPt, axisEta});
80  histos.add("CovMat_cSnpZ", "cSnpZ" + commonTitle, kTProfile2D, {axisPt, axisEta});
81  histos.add("CovMat_cSnpSnp", "cSnpSnp" + commonTitle, kTProfile2D, {axisPt, axisEta});
82  histos.add("CovMat_cTglY", "cTglY" + commonTitle, kTProfile2D, {axisPt, axisEta});
83  histos.add("CovMat_cTglZ", "cTglZ" + commonTitle, kTProfile2D, {axisPt, axisEta});
84  histos.add("CovMat_cTglSnp", "cTglSnp" + commonTitle, kTProfile2D, {axisPt, axisEta});
85  histos.add("CovMat_cTglTgl", "cTglTgl" + commonTitle, kTProfile2D, {axisPt, axisEta});
86  histos.add("CovMat_c1PtY", "c1PtY" + commonTitle, kTProfile2D, {axisPt, axisEta});
87  histos.add("CovMat_c1PtZ", "c1PtZ" + commonTitle, kTProfile2D, {axisPt, axisEta});
88  histos.add("CovMat_c1PtSnp", "c1PtSnp" + commonTitle, kTProfile2D, {axisPt, axisEta});
89  histos.add("CovMat_c1PtTgl", "c1PtTgl" + commonTitle, kTProfile2D, {axisPt, axisEta});
90  histos.add("CovMat_c1Pt21Pt2", "c1Pt21Pt2" + commonTitle, kTProfile2D, {axisPt, axisEta});
91 
92  histos.add("Efficiency", "Efficiency" + commonTitle, kTProfile2D, {axisPt, axisEta});
93  }

95  void process(const soa::Join<aod::Tracks, aod::TracksCov, aod::McTrackLabels>& tracks,
96  const aod::McParticles& mcParticles)
97  {
98  std::vector<int64_t> recoTracks(tracks.size());
99  int ntrks = 0;
100 
101  for (const auto& track : tracks) {
102  const auto mcParticle = track.mcParticle();
103  if (mcParticle.pdgCode() != pdg) {
104  continue;
105  }
106  if (selPrim.value && !MC::isPhysicalPrimary(mcParticles, mcParticle)) { // Requiring is physical primary
107  continue;
108  }
109 
110  recoTracks[ntrks++] = mcParticle.globalIndex();
111 
112  histos.fill(HIST("pt"), mcParticle.pt());
113  histos.fill(HIST("eta"), mcParticle.eta());
114  histos.fill(HIST("CovMat_cYY"), mcParticle.pt(), mcParticle.eta(), track.cYY());
115  histos.fill(HIST("CovMat_cZY"), mcParticle.pt(), mcParticle.eta(), track.cZY());
116  histos.fill(HIST("CovMat_cZZ"), mcParticle.pt(), mcParticle.eta(), track.cZZ());
117  histos.fill(HIST("CovMat_cSnpY"), mcParticle.pt(), mcParticle.eta(), track.cSnpY());
118  histos.fill(HIST("CovMat_cSnpZ"), mcParticle.pt(), mcParticle.eta(), track.cSnpZ());
119  histos.fill(HIST("CovMat_cSnpSnp"), mcParticle.pt(), mcParticle.eta(), track.cSnpSnp());
120  histos.fill(HIST("CovMat_cTglY"), mcParticle.pt(), mcParticle.eta(), track.cTglY());
121  histos.fill(HIST("CovMat_cTglZ"), mcParticle.pt(), mcParticle.eta(), track.cTglZ());
122  histos.fill(HIST("CovMat_cTglSnp"), mcParticle.pt(), mcParticle.eta(), track.cTglSnp());
123  histos.fill(HIST("CovMat_cTglTgl"), mcParticle.pt(), mcParticle.eta(), track.cTglTgl());
124  histos.fill(HIST("CovMat_c1PtY"), mcParticle.pt(), mcParticle.eta(), track.c1PtY());
125  histos.fill(HIST("CovMat_c1PtZ"), mcParticle.pt(), mcParticle.eta(), track.c1PtZ());
126  histos.fill(HIST("CovMat_c1PtSnp"), mcParticle.pt(), mcParticle.eta(), track.c1PtSnp());
127  histos.fill(HIST("CovMat_c1PtTgl"), mcParticle.pt(), mcParticle.eta(), track.c1PtTgl());
128  histos.fill(HIST("CovMat_c1Pt21Pt2"), mcParticle.pt(), mcParticle.eta(), track.c1Pt21Pt2());
129  }
130  histos.fill(HIST("multiplicity"), ntrks);
131 
132  for (const auto& mcParticle : mcParticles) {
133  if (mcParticle.pdgCode() != pdg) {
134  continue;
135  }
136  if (!MC::isPhysicalPrimary(mcParticles, mcParticle)) { // Requiring is physical primary
137  continue;
138  }
139 
140  if (std::find(recoTracks.begin(), recoTracks.end(), mcParticle.globalIndex()) != recoTracks.end()) {
141  histos.fill(HIST("Efficiency"), mcParticle.pt(), mcParticle.eta(), 1.);
142  } else {
143  histos.fill(HIST("Efficiency"), mcParticle.pt(), mcParticle.eta(), 0.);
144  }
145  }
146  }
147 };
