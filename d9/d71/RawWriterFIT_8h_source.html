60  void convertDigitsToRaw(const std::string& outputDir, const std::string& filenameDigits)
61  {
62  LOG(INFO) << "Converting Digits to Raw data...";
63  mWriter.setCarryOverCallBack(this);
64  LookupTable_t::Instance().printFullMap();
65  //Preparing topo2FEEmetadata map
66  mMapTopo2FEEmetadata.clear();
67  mMapTopo2FEEmetadata = LookupTable_t::Instance().template makeMapFEEmetadata<o2::header::RAWDataHeader, RDHUtils>();
68  //Preparing filenames
69  std::string detNameLowCase = LookupTable_t::sDetectorName;
70  std::for_each(detNameLowCase.begin(), detNameLowCase.end(), [](char& c) { c = ::tolower(c); });
71  auto makeFilename = [&](const o2::header::RAWDataHeader& rdh) -> std::string {
72  std::string maskName = detNameLowCase + "_link";
73  std::string outputFilename = mOutputPerLink ? o2::utils::Str::concat_string(outputDir, maskName, std::to_string(RDHUtils::getFEEID(rdh)), ".raw") : o2::utils::Str::concat_string(outputDir, detNameLowCase + ".raw");
74  return outputFilename;
75  };
76  //Registering links
77  for (const auto& metadataPair : mMapTopo2FEEmetadata) {
78  const auto& rdh = metadataPair.second;
79  const auto outputFilename = makeFilename(rdh);
80  mWriter.registerLink(RDHUtils::getFEEID(rdh), RDHUtils::getCRUID(rdh), RDHUtils::getLinkID(rdh), RDHUtils::getEndPointID(rdh), outputFilename);
81  }
82  //Processing digits into raw data
83  TFile* inputFile = TFile::Open(filenameDigits.c_str());
84  assert(inputFile != nullptr);
85  LOG(INFO) << "Source file: " << filenameDigits;
86  TTree* inputTree = dynamic_cast<TTree*>(inputFile->Get("o2sim"));
87  DigitBlockFIT_t::processDigitBlocks(inputTree, *this);
88  delete inputTree;
89  inputFile->Close();
90  delete inputFile;
91  }
92  void processDigitBlockPerTF(const std::vector<DigitBlockFIT_t>& vecDigitBlock) // Is used in DigitBlockFIT_t::processDigitBlocks for each TF (TTree entry)
93  {
94  for (const auto& digitBlock : vecDigitBlock) {
95  //Processing PM data
96  auto mapDataBlockPM = digitBlock.template decomposeDigits<DataBlockPM_t>();
97  if (mVerbosity > 0) {
98  digitBlock.print();
99  }
100  for (const auto& dataBlockPair : mapDataBlockPM) {
101  const auto& topo = dataBlockPair.first;
102  const auto& dataBlock = dataBlockPair.second;
103  const auto itRdh = mMapTopo2FEEmetadata.find(topo);
104  assert(itRdh != mMapTopo2FEEmetadata.end());
105  if (mVerbosity > 0) {
106  dataBlock.print();
107  }
108  const auto& rdh = itRdh->second;
109  auto data = dataBlock.serialize();
110  mWriter.addData(RDHUtils::getFEEID(rdh), RDHUtils::getCRUID(rdh), RDHUtils::getLinkID(rdh), RDHUtils::getEndPointID(rdh), dataBlock.getInteractionRecord(), data);
111  }
112  //Processing TCM data
113  const auto dataBlockPair = digitBlock.template decomposeDigits<DataBlockTCM_t>();
114  const auto& topo = dataBlockPair.first;
115  const auto& dataBlock = dataBlockPair.second;
116  const auto itRdh = mMapTopo2FEEmetadata.find(topo);
117  assert(itRdh != mMapTopo2FEEmetadata.end());
118  if (mVerbosity > 0) {
119  dataBlock.print();
120  }
121  const auto& rdh = itRdh->second;
122  auto data = dataBlock.serialize();
123  mWriter.addData(RDHUtils::getFEEID(rdh), RDHUtils::getCRUID(rdh), RDHUtils::getLinkID(rdh), RDHUtils::getEndPointID(rdh), dataBlock.getInteractionRecord(), data);
124  }
125  }
