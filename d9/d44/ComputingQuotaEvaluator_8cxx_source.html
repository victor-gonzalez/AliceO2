57 bool ComputingQuotaEvaluator::selectOffer(int task, ComputingQuotaRequest const& selector)
58 {
59  auto selectOffer = [&loop = mLoop, &offers = this->mOffers, &infos = this->mInfos, task](int ref) {
60  auto& selected = offers[ref];
61  auto& info = infos[ref];
62  selected.user = task;
63  if (info.firstUsed == 0) {
64  info.firstUsed = uv_now(loop);
65  }
66  info.lastUsed = uv_now(loop);
67  };
68 
69  ComputingQuotaOffer accumulated;
70  static QuotaEvaluatorStats stats;
71 
72  stats.invalidOffers.clear();
73  stats.otherUser.clear();
74  stats.unexpiring.clear();
75  stats.selectedOffers.clear();
76  stats.expired.clear();
77 
78  auto summarizeWhatHappended = [](bool enough, std::vector<int> const& result, ComputingQuotaOffer const& totalOffer, QuotaEvaluatorStats& stats) -> bool {
79  if (result.size() == 1 && result[0] == 0) {
80  // LOG(INFO) << "No particular resource was requested, so we schedule task anyways";
81  return enough;
82  }
83  if (enough) {
84  LOGP(INFO, "{} offers were selected for a total of: cpu {}, memory {}, shared memory {}", result.size(), totalOffer.cpu, totalOffer.memory, totalOffer.sharedMemory);
85  LOGP(INFO, " The following offers were selected for computation: {} ", fmt::join(result, ","));
86  } else {
87  LOG(INFO) << "No offer was selected";
88  if (result.size()) {
89  LOGP(INFO, " The following offers were selected for computation but not enough: {} ", fmt::join(result, ","));
90  }
91  }
92  if (stats.invalidOffers.size()) {
93  LOGP(INFO, " The following offers were invalid: {}", fmt::join(stats.invalidOffers, ", "));
94  }
95  if (stats.otherUser.size()) {
96  LOGP(INFO, " The following offers were owned by other users: {}", fmt::join(stats.otherUser, ", "));
97  }
98  if (stats.expired.size()) {
99  LOGP(INFO, " The following offers are expired: {}", fmt::join(stats.expired, ", "));
100  }
101  if (stats.unexpiring.size() > 1) {
102  LOGP(INFO, " The following offers will never expire: {}", fmt::join(stats.unexpiring, ", "));
103  }
104 
105  return enough;
106  };
107 
108  bool enough = false;
109 
110  for (int i = 0; i != mOffers.size(); ++i) {
111  auto& offer = mOffers[i];
112  auto& info = mInfos[i];
113 
114  // Ignore:
115  // - Invalid offers
116  // - Offers which belong to another task
117  // - Expired offers
118  if (offer.valid == false) {
119  stats.invalidOffers.push_back(i);
120  continue;
121  }
122  if (offer.user != -1 && offer.user != task) {
123  stats.otherUser.push_back(i);
124  continue;
125  }
126  if (offer.runtime < 0) {
127  stats.unexpiring.push_back(i);
128  } else if (offer.runtime + info.received < uv_now(mLoop)) {
129  LOGP(INFO, "Offer {} expired since {} milliseconds and holds {}MB", i, uv_now(mLoop) - offer.runtime - info.received, offer.sharedMemory / 1000000);
130  mExpiredOffers.push_back(ComputingQuotaOfferRef{i});
131  stats.expired.push_back(i);
132  continue;
133  } else {
134  LOGP(INFO, "Offer {} still valid for {} milliseconds, providing {}MB", i, offer.runtime + info.received - uv_now(mLoop), offer.sharedMemory / 1000000);
135  }
137  assert(offer.sharedMemory >= 0);
138  auto tmp = accumulated;
139  tmp.cpu += offer.cpu;
140  tmp.memory += offer.memory;
141  tmp.sharedMemory += offer.sharedMemory;
142  offer.score = selector(offer, tmp);
143  switch (offer.score) {
144  case OfferScore::Unneeded:
145  continue;
146  case OfferScore::Unsuitable:
147  continue;
148  case OfferScore::More:
149  selectOffer(i);
150  accumulated = tmp;
151  stats.selectedOffers.push_back(i);
152  continue;
153  case OfferScore::Enough:
154  selectOffer(i);
155  accumulated = tmp;
156  stats.selectedOffers.push_back(i);
157  enough = true;
158  break;
159  };
160  }
161  // If we get here it means we never got enough offers, so we return false.
162  return summarizeWhatHappended(enough, stats.selectedOffers, accumulated, stats);
163 }

165 void ComputingQuotaEvaluator::consume(int id, ComputingQuotaConsumer& consumer)
166 {
167  using o2::monitoring::Metric;
168  using o2::monitoring::Monitoring;
169  using o2::monitoring::tags::Key;
170  using o2::monitoring::tags::Value;
171  // This will report how much of the offers has to be considered consumed.
172  // Notice that actual memory usage might be larger, because we can over
173  // allocate.
174  auto reportConsumedOffer = [&totalDisposedMemory = mTotalDisposedSharedMemory, &monitoring = mRegistry.get<Monitoring>()](ComputingQuotaOffer const& accumulatedConsumed) {
175  totalDisposedMemory += accumulatedConsumed.sharedMemory;
176  monitoring.send(Metric{(uint64_t)totalDisposedMemory, "shm-offer-consumed"}.addTag(Key::Subsystem, Value::DPL));
177  };
178  consumer(id, mOffers, reportConsumedOffer);
179 }

181 void ComputingQuotaEvaluator::dispose(int taskId)
182 {
183  for (int oi = 0; oi < mOffers.size(); ++oi) {
184  auto& offer = mOffers[oi];
185  if (offer.user != taskId) {
186  continue;
187  }
188  offer.user = -1;
189  // Disposing the offer so that the resource can be recyled.
192  if (oi == 0) {
193  return;
194  }
195  if (offer.valid == false) {
196  continue;
197  }
198  if (offer.sharedMemory <= 0) {
199  offer.valid = false;
200  offer.score = OfferScore::Unneeded;
201  }
202  }
203 }

206 void ComputingQuotaEvaluator::updateOffers(std::vector<ComputingQuotaOffer>& pending)
207 {
208  for (size_t oi = 0; oi < mOffers.size(); oi++) {
209  auto& storeOffer = mOffers[oi];
210  auto& info = mInfos[oi];
211  if (pending.empty()) {
212  return;
213  }
214  if (storeOffer.valid == true) {
215  continue;
216  }
217  info.received = uv_now(mLoop);
218  auto& offer = pending.back();
219  storeOffer = offer;
220  pending.pop_back();
221  }
222 }

224 void ComputingQuotaEvaluator::handleExpired()
225 {
226  static int nothingToDoCount = mExpiredOffers.size();
227  if (mExpiredOffers.size()) {
228  LOGP(INFO, "Handling {} expired offers", mExpiredOffers.size());
229  nothingToDoCount = 0;
230  } else {
231  if (nothingToDoCount == 0) {
232  nothingToDoCount++;
233  LOGP(INFO, "No expired offers");
234  }
235  }
236  using o2::monitoring::Metric;
237  using o2::monitoring::Monitoring;
238  using o2::monitoring::tags::Key;
239  using o2::monitoring::tags::Value;
240  auto& monitoring = mRegistry.get<o2::monitoring::Monitoring>();
243  static uint64_t expiredOffers = 0;
244  static uint64_t expiredBytes = 0;
245 
246  for (auto& ref : mExpiredOffers) {
247  auto& offer = mOffers[ref.index];
248  if (offer.sharedMemory < 0) {
249  LOGP(INFO, "Offer {} does not have any more memory. Marking it as invalid.", ref.index);
250  offer.valid = false;
251  offer.score = OfferScore::Unneeded;
252  continue;
253  }
254  // FIXME: offers should go through the driver client, not the monitoring
255  // api.
256  auto& monitoring = mRegistry.get<o2::monitoring::Monitoring>();
257  monitoring.send(o2::monitoring::Metric{expiredOffers++, "resource-offer-expired"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
258  expiredBytes += offer.sharedMemory;
259  monitoring.send(o2::monitoring::Metric{(uint64_t)expiredBytes, "arrow-bytes-expired"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
260  LOGP(INFO, "Offer {} expired. Giving back {}MB and {} cores", ref.index, offer.sharedMemory / 1000000, offer.cpu);
261  //driverClient.tell("expired shmem {}", offer.sharedMemory);
262  //driverClient.tell("expired cpu {}", offer.cpu);
263  offer.sharedMemory = -1;
264  offer.valid = false;
265  offer.score = OfferScore::Unneeded;
266  }
267  mExpiredOffers.clear();
268 }
