32 float o2::tpc::RobustAverage::getStdDev(const float mean) const
33 {
34  std::vector<float> diff(mValues.size());
35  std::transform(mValues.begin(), mValues.end(), diff.begin(), [mean](const float val) { return val - mean; });
36  const float sqsum = std::inner_product(diff.begin(), diff.end(), diff.begin(), decltype(diff)::value_type(0));
37  const float stdev = std::sqrt(sqsum / diff.size());
38  return stdev;
39 }

41 void o2::tpc::RobustAverage::filterOutliers(const float mean, const float stdev, const float sigma)
42 {
43  std::sort(mValues.begin(), mValues.end());
44  const float sigmastddev = sigma * stdev;
45  const float minVal = mean - sigmastddev;
46  const float maxVal = mean + sigmastddev;
47  const auto upper = std::upper_bound(mValues.begin(), mValues.end(), maxVal);
48  mValues.erase(upper, mValues.end());
49  const auto lower = std::lower_bound(mValues.begin(), mValues.end(), minVal);
50  mValues.erase(mValues.begin(), lower);
51 }
